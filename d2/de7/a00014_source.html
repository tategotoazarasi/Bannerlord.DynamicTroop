37 private static int CalculateEquipmentValue(this CharacterObject? character) {
38 if (character == null) return 0;
39
40 var totalValue = 0;
41 IEnumerable<Equipment> equipments = character.BattleEquipments;
42 if (equipments == null) return 0;
43
44 var cnt = character.BattleEquipments.Count();
45 foreach (var equipment in equipments) {
46 foreach (var slot in Global.EquipmentSlots) {
47 var element = equipment.GetEquipmentFromSlot(slot);
48 if (element.IsEmpty || element.Item == null) continue;
49
50 totalValue += equipment.GetEquipmentFromSlot(slot).ItemValue;
51 }
52 }
53
54 return totalValue / cnt;
55 }
