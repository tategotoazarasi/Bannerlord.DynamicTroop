53 private static int BinarySearchClosest(List<float> sortedValues, double value) {
54 int left = 0, right = sortedValues.Count - 1;
55 while (left < right) {
56 var mid = (left + right) / 2;
57 var midValue = sortedValues[mid];
58 if (midValue < value) { left = mid; }
59 else if (midValue > value) { right = mid; }
60 else { return mid; }
61

63 }
64
