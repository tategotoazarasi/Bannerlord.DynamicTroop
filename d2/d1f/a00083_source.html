68 private static void Postfix(Agent __result, SpawnAgentState? __state) {

71 var equipmentToConsume = __state.Assignment.CreateEquipmentForArmoryConsumption();

83 dynamicTroopMissionLogic?.RegisterSpawnedAgentAssignment(__result, __state.Assignment);
84 }

96 private static bool IsPartyValidForProcessing(IAgentOriginBase agentOrigin) {
97 var party = agentOrigin switch {
98 PartyAgentOrigin partyAgentOrigin => partyAgentOrigin.Party,
99 PartyGroupAgentOrigin partyGroupAgentOrigin => partyGroupAgentOrigin.Party,
100 SimpleAgentOrigin simpleAgentOrigin => simpleAgentOrigin.Party,
101 _ => null
102 };
103
104 if (party is not { IsValid: true, MobileParty: not null }) { return false; }
