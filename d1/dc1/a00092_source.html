39 var canOpenArmory = mapBar != null && CanOpenArmory(mapBar, mainParty);
40
41 IsArmoryReadinessVisible = canOpenArmory;
42 if (!canOpenArmory || mainParty == null)

44 if (!ReferenceEquals(_trackedMainParty, mainParty)) {
45 _trackedMainParty = mainParty;

47 _lastMemberRosterVersion = -1;
48 _hasReadinessContextSnapshot = false;

51
52 var armoryVersion = ArmyArmory.Armory.VersionNo;
53 var memberRosterVersion = mainParty.MemberRoster.VersionNo;
54 var preferDefaultEquipmentThenClosest = ModSettings.Instance?.PreferDefaultEquipmentThenClosest ?? true;
55 var mainPartyAtSea = mainParty.IsCurrentlyAtSea;
56 var readinessContextChanged = !_hasReadinessContextSnapshot ||
57 _lastPreferDefaultEquipmentThenClosest != preferDefaultEquipmentThenClosest ||
58 _lastMainPartyAtSea != mainPartyAtSea;
59 var troopCompositionChanged = false;
60
61 if (_lastMemberRosterVersion != memberRosterVersion) {

63 troopCompositionChanged = MainPartyTroopCompositionChanged(mainParty);

66 if (_lastArmoryVersion == armoryVersion && !troopCompositionChanged && !readinessContextChanged)
67 return;
68
69 _lastArmoryVersion = armoryVersion;
70 _hasReadinessContextSnapshot = true;
71 _lastPreferDefaultEquipmentThenClosest = preferDefaultEquipmentThenClosest;
72 _lastMainPartyAtSea = mainPartyAtSea;
73
74 var readiness = PartyEquipmentDistributor.MeasureMainPartyArmoryReadiness(!mainPartyAtSea);
75 ArmoryReadinessFillHeight = INDICATOR_HEIGHT * readiness.FillRatio;
76
77 var hintText = new TextObject("{=armory_readiness_hint}Army Armory readiness: {PERCENT}% ({FILLED}/{EXPECTED} required equipment slots filled)");
78 hintText.SetTextVariable("PERCENT", readiness.Percentage);

84 private bool MainPartyTroopCompositionChanged(MobileParty mainParty) {
85 var currentTroopCounts = new Dictionary<CharacterObject, int>();
86
87 foreach (var rosterElement in mainParty.MemberRoster.GetTroopRoster()) {
88 if (rosterElement.Character is CharacterObject { IsHero: false } character && rosterElement.Number > 0)
89 currentTroopCounts[character] = rosterElement.Number;
90 }
91
92 if (currentTroopCounts.Count == _lastMainPartyTroopCounts.Count) {
93 var compositionChanged = false;
94 foreach (var troopCount in currentTroopCounts) {
95 if (!_lastMainPartyTroopCounts.TryGetValue(troopCount.Key, out var previousCount) || previousCount != troopCount.Value) {
96 compositionChanged = true;
97 break;
98 }
99 }
100
101 if (!compositionChanged)
102 return false;
103 }
104
105 _lastMainPartyTroopCounts.Clear();
106 foreach (var troopCount in currentTroopCounts)
107 _lastMainPartyTroopCounts.Add(troopCount.Key, troopCount.Value);
108
109 return true;
110 }
