48 public static void AssignEquipment(Equipment equipment) {
49 foreach (var slot in Global.EquipmentSlots) {
50 var element = equipment.GetEquipmentFromSlot(slot);
51
52 // 使用模式匹配来检查条件，并反转if语句来减少嵌套
53 if (element.IsEmpty || element.Item is null) continue;
54
55 var itemToAssign =
56 Armory.FirstOrDefaultQ(a => !a.IsEmpty &&
57 a.EquipmentElement.Item.StringId == element.Item.StringId &&
58 a.Amount > 0);
59
60 if (!itemToAssign.IsEmpty)
61 _ = Armory.AddToCounts(itemToAssign.EquipmentElement, -1);
62 else
63 Global.Log($"Assigning Empty item {element.Item.StringId}", Colors.Red, Level.Warn);
64 }

90 private static int SellExcessEquipment() {
91 RebuildArmory();
92 var excessValue = 0;
93 var playerParty = MobileParty.MainParty;
94 if (playerParty?.MemberRoster?.GetTroopRoster() == null) return 0;
95
96 var memberCnt = playerParty.MemberRoster.GetTroopRoster()
97 .WhereQ(element => element.Character is { IsHero: false })
98 .SumQ(element => element.Number);
99
100 foreach (var equipmentAndThreshold in EveryoneCampaignBehavior.EquipmentAndThresholds) {
101 var armorTotalCount = Armory.WhereQ(kv => kv.EquipmentElement.Item?.ItemType == equipmentAndThreshold.Key)
102 .SumQ(kv => kv.Amount);
103 var surplusCount = armorTotalCount - equipmentAndThreshold.Value(memberCnt);
104 if (surplusCount <= 0) continue;
105
106 var surplusCountCpy = surplusCount;
107
108 // 创建优先级队列
109 ItemPriorityQueue armorQueue = new(new ArmorElementComparer());
110 foreach (var kv in Armory.WhereQ(kv => kv.EquipmentElement.Item?.ItemType == equipmentAndThreshold.Key))
111 armorQueue.Enqueue(kv.EquipmentElement, kv.EquipmentElement.ItemValue);
112
113 // 移除多余的装备
114 while (surplusCount > 0 && !armorQueue.IsEmpty) {
115 var lowestArmor = armorQueue.Dequeue();
116 var countToRemove = Math.Min(Armory.GetElementNumber(Armory.FindIndexOfElement(lowestArmor.Key)),
117 surplusCount);
118
119 //Global.Debug($"countToRemove={countToRemove}, lowestArmorNumber={Armory.GetItemNumber(lowestArmor.Key)}");
120 _ = Armory.AddToCounts(lowestArmor.Key, -countToRemove); // 减少数量
121 surplusCount -= countToRemove;
122 excessValue += countToRemove * lowestArmor.Key.ItemValue;
123 Global.Debug($"Sold {countToRemove}x{lowestArmor.Key.ItemValue} from player's armory");
124 }
125
126 Global.Debug($"Sold {surplusCountCpy - surplusCount}x{equipmentAndThreshold.Key} items from player's armory");
127 }
128
129 Global.Debug($"Sold {excessValue} denars worth of equipment from player's armory");
130 return excessValue;
131 }
