36 private static int CalculateEquipmentValue(this CharacterObject? character) {
37 if (character == null) return 0;
38
39 var totalValue = 0;
40 IEnumerable<Equipment> equipments = character.BattleEquipments;
41 if (equipments == null) return 0;
42
43 var cnt = character.BattleEquipments.Count();
44 foreach (var equipment in equipments) {
45 foreach (var slot in Global.EquipmentSlots) {
46 var element = equipment.GetEquipmentFromSlot(slot);
47 if (element.IsEmpty || element.Item == null) continue;
48
49 totalValue += equipment.GetEquipmentFromSlot(slot).ItemValue;
50 }
51 }
52
53 return totalValue / cnt;
54 }

80 public static void Init() {
81 var cultureList = MBObjectManager.Instance.GetObjectTypeList<CultureObject>();
82 _cachedSkillValue = MBObjectManager.Instance.GetObjectTypeList<SkillObject>();
83 foreach (var culture in cultureList) {
84 Enqueue(culture?.EliteBasicTroop, TroopType.Elite);
85
86 MBReadOnlyList<CharacterObject>? basicMercenaryTroops = culture?.BasicMercenaryTroops;
87 if (basicMercenaryTroops != null)
88 foreach (var troop in basicMercenaryTroops)
89 Enqueue(troop, TroopType.Mercenary);
90
91 Enqueue(culture?.BasicTroop, TroopType.Basic);
92
93 Enqueue(culture?.BanditChief, TroopType.Bandit);
94 Enqueue(culture?.BanditRaider, TroopType.Bandit);
95 Enqueue(culture?.BanditBandit, TroopType.Bandit);
96 Enqueue(culture?.BanditBoss, TroopType.Bandit);
97
98 Enqueue(culture?.MeleeMilitiaTroop, TroopType.Militia);
99 Enqueue(culture?.MeleeEliteMilitiaTroop, TroopType.Militia);
100 Enqueue(culture?.RangedEliteMilitiaTroop, TroopType.Militia);
101 Enqueue(culture?.RangedMilitiaTroop, TroopType.Militia);
102 Enqueue(culture?.MilitiaArcher, TroopType.Militia);
103 Enqueue(culture?.MilitiaVeteranArcher, TroopType.Militia);
104 Enqueue(culture?.MilitiaSpearman, TroopType.Militia);
105 Enqueue(culture?.MilitiaVeteranSpearman, TroopType.Militia);
106 }
107
108 while (!TroopQueue.IsEmpty()) {
109 (var character, var typ) = TroopQueue.Dequeue();
110 if (TypeDict.ContainsKey(character)) continue;
111
112 TypeDict.Add(character, typ);
113 foreach (var next in character.UpgradeTargets) Enqueue(next, typ);
114 }
115 }
