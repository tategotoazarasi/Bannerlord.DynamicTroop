97 public int CompareTo(object? obj) {
98 if (obj == null) return 1;
99
100 if (obj is not Assignment other) throw new ArgumentException("Object is not an Assignment");
101
102 var tierComparison = Character.Tier.CompareTo(other.Character.Tier);
103 if (tierComparison != 0) return tierComparison;
104
105 // TroopType 比较，小的放在前面
106 var thisTroopType = Character.GetTroopType();
107 var otherTroopType = other.Character.GetTroopType();
108 var troopTypeComparison = thisTroopType.CompareTo(otherTroopType);
109 if (troopTypeComparison != 0) return troopTypeComparison;
110
111 // 按 IsRanged 比较，IsRanged 排在 !IsRanged 前面
112 if (!Character.IsRanged && other.Character.IsRanged) return 1;
113
114 if (Character.IsRanged && !other.Character.IsRanged) return -1;
115
116 // 按 IsMounted 比较，!IsMounted 排在 IsMounted 前面
117 if (Character.IsMounted && !other.Character.IsMounted) return 1;
118
119 if (!Character.IsMounted && other.Character.IsMounted) return -1;
120
121 var skillValueComparison = Character.SkillValue().CompareTo(other.Character.SkillValue());
122 if (skillValueComparison != 0) return skillValueComparison;
123
124 var equipmentValueComparison = Character.EquipmentValue().CompareTo(other.Character.EquipmentValue());
125 if (equipmentValueComparison != 0) return equipmentValueComparison;
126
127 var levelComparison = Character.Level.CompareTo(other.Character.Level);
128 if (levelComparison != 0) return levelComparison;
129
130 return 0; // 如果所有条件都相等，则认为两者相等
131 }

141 public void FillEmptySlots() {
142 foreach (var slot in Global.ArmourSlots) {
143 var referenceEquipment = ReferenceEquipment.GetEquipmentFromSlot(slot);
144 if (Equipment.GetEquipmentFromSlot(slot) is not { IsEmpty: false, Item: not null }) {
145 var itemType = Helper.EquipmentIndexToItemEnumType(slot);
146 if (!itemType.HasValue) continue;
147
148 ItemObject? item;
149 if (referenceEquipment is { IsEmpty: false, Item: not null }) {
150 List<ItemObject> itemList = new() { referenceEquipment.Item };
151 var itemsByCharacter =
152 Cache.GetItemsByTypeTierAndCulture(itemType.Value, Character.Tier, Character.Culture);
153 if (itemsByCharacter != null) itemList.AddRange(itemsByCharacter);
154
155 if (referenceEquipment.Item.Culture is CultureObject cultureObject) {
156 var itemsByReference =
157 Cache.GetItemsByTypeTierAndCulture(itemType.Value,
158 (int)referenceEquipment.Item.Tier,
159 cultureObject);
160 if (itemsByReference != null) itemList.AddRange(itemsByReference);
161 }
162
163 item = WeightedRandomSelector.SelectItem(itemList, referenceEquipment.Item.Effectiveness);
164 }
165 else {
166 item = Cache.GetItemsByTypeTierAndCulture(itemType.Value, Character.Tier, Character.Culture)
167 ?.GetRandomElement();
168 if (item == null) continue;
169 }
170
171 Equipment.AddEquipmentToSlotWithoutAgent(slot, new EquipmentElement(item));
172 }
173 }
174
175 foreach (var slot in Global.EquipmentSlots)
176 if (Equipment.GetEquipmentFromSlot(slot) is not { IsEmpty: false, Item: not null })
177 Equipment.AddEquipmentToSlotWithoutAgent(slot, ReferenceEquipment.GetEquipmentFromSlot(slot));
178 }
