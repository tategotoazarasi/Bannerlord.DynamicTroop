53 public PartyEquipmentDistributor(Mission mission, MobileParty party, Dictionary<ItemObject, int> objectToAssign) {
54 _mission = mission;
55 _party = party;
56 _itemRoster = null;
57 _equipmentToAssign = new ConcurrentDictionary<EquipmentElement, int>(new EquipmentElementComparer());
58
59 foreach (var kv in objectToAssign) {
60 if (kv.Key != null) {
61 EquipmentElement element = new(kv.Key);
62
63 if (_equipmentToAssign.TryGetValue(element, out var existingCount)) { _equipmentToAssign[element] = existingCount + kv.Value; }
64 else { _ = _equipmentToAssign.TryAdd(element, kv.Value); }
65 }

69 public void RunAsync() {
70 foreach (var troop in _party.MemberRoster.GetTroopRoster()) {
71 for (var i = 0; i < troop.Number - troop.WoundedNumber; i++) {
72 if (troop.Character is CharacterObject characterObject && !characterObject.IsHero) { Assignments.Add(new Assignment(characterObject)); }
73 }
74 }
75
76 Assignments.Sort((x, y) => y.CompareTo(x));
77 if (_itemRoster != null) {
78 foreach (var kv in _itemRoster) {
79 if (kv is not { IsEmpty: false, EquipmentElement.IsEmpty: false }) { continue; }
80
81 // 尝试获取已存在的数量
82 if (!_equipmentToAssign.TryGetValue(kv.EquipmentElement, out var existingAmount)) {
83 // 如果键不存在，添加新的键值对
84 _ = _equipmentToAssign.TryAdd(kv.EquipmentElement, kv.Amount);
85 }
86 else {
87 // 如果键已存在，更新数量
88 _equipmentToAssign[kv.EquipmentElement] = existingAmount + kv.Amount;
89 }
90 }
91 }
92
93 DoAssignAsync();
94 }

113 private static void PopulateDictionaries(IEnumerable<KeyValuePair<EquipmentElement, int>> equipment,
114 IDictionary<int, List<(EquipmentElement Key, int Cnt)>> horsesDict,
115 IDictionary<int, List<(EquipmentElement Key, int Cnt)>> harnessDict,
116 ICollection<(EquipmentElement Key, int Cnt)> saddles) {
117 foreach (var kvp in equipment) {
118 switch (kvp.Key.Item) {
119 case { HasHorseComponent: true, ItemType: ItemTypeEnum.Horse }:
120 AddToDict(horsesDict, kvp.Key, kvp.Value);

122
123 case { HasArmorComponent: true, ItemType: ItemTypeEnum.HorseHarness }:
124 AddToDict(harnessDict, kvp.Key, kvp.Value);

126
127 case {
128 HasArmorComponent : false,
129 HasSaddleComponent: true,
130 ItemType : ItemTypeEnum.HorseHarness
131 }:
132 saddles.Add((kvp.Key, kvp.Value));
133 break;
134 }
135 }

164 private void GenerateHorseAndHarnessPairs(Dictionary<int, List<(EquipmentElement Key, int Cnt)>> horsesDict, IDictionary<int, List<(EquipmentElement Key, int Cnt)>> harnessDict, IList<(EquipmentElement Key, int Cnt)> saddles) {
165 foreach (var kvp in horsesDict) {
166 var familyType = kvp.Key;
167 var horses = kvp.Value.ToArrayQ();
168 var harnesses = harnessDict.TryGetValue(familyType, out var harnessesList) ? harnessesList.ToArrayQ() : Array.Empty<(EquipmentElement Key, int Cnt)>();
169
170 int horseIndex = 0, harnessIndex = 0;
171 while (horseIndex < horses.Length) {
172 (var horseItem, var horseCnt) = horses[horseIndex];
173 var harnessItem = harnessIndex < harnesses.Length ? harnesses[harnessIndex].Key : new EquipmentElement(null);
174 var harnessCnt = harnessIndex < harnesses.Length ? harnesses[harnessIndex].Cnt : 0;
175
176 if (horseCnt > 0) {
177 HorseAndHarness hah = new(horseItem, harnessItem);
178 _horseAndHarnesses.Add(hah);
179 horses[horseIndex] = (horseItem, --horseCnt);
180
181 if (harnessCnt > 0) { harnesses[harnessIndex] = (harnessItem, --harnessCnt); }
182 }
183
184 if (horseCnt == 0) { horseIndex++; }
185
186 if (harnessCnt == 0 && harnessIndex < harnesses.Length) { harnessIndex++; }
187 }
188 }
189
190 AssignSaddlesToHorseAndHarnesses(_horseAndHarnesses, saddles);
191 _horseAndHarnesses.Sort((x, y) => y.CompareTo(x));
192 }

194 private static void AssignSaddlesToHorseAndHarnesses(IEnumerable<HorseAndHarness> horseAndHarnesses, IList<(EquipmentElement Key, int Cnt)> saddles) {
195 var saddleIndex = 0;

204 if (saddleCnt == 0) { saddleIndex++; }

288 var candidates = new List<KeyValuePair<EquipmentElement, int>>();
289
290 foreach (var kv in _equipmentToAssign) {
291 if (kv.Key is { IsEmpty: false, Item: not null } &&
292 kv.Value > 0 &&
293 equipmentFilter(kv)) { candidates.Add(kv); }

295
296 if (candidates.Count == 0)
297 return;
298
299 candidates.Sort(static (a, b) => {
300 var tierCompare = b.Key.Item.Tier.CompareTo(a.Key.Item.Tier);
301 return tierCompare != 0 ? tierCompare : b.Key.Item.Value.CompareTo(a.Key.Item.Value);

312 continue;
313
314 while (currentIndex < candidates.Count && candidates[currentIndex].Value <= 0)
315 currentIndex++;
316

318 return;

320 var current = candidates[currentIndex];

322 assignment.SetEquipment(slot.Value, current.Key);
323 Global.Debug($"extra equipment {current.Key} assigned to {assignment.Character.StringId}#{assignment.Index} on slot {slot.Value} for {_party.Name}");

326 ConsumeEquipmentToAssign(current.Key);

405 private EquipmentElement? GetOneRandomMeleeWeapon(Assignment assignment) {
406 var candidates = new List<EquipmentElement>();
407
408 foreach (var kv in _equipmentToAssign) {
409 var element = kv.Key;
410 if (kv.Value <= 0 || element.IsEmpty || element.Item == null)
411 continue;
412
413 var item = element.Item;
414 if (!item.HasWeaponComponent)
415 continue;
416
417 if (!item.IsSuitableForCharacter(assignment.Character))
418 continue;
419
420 if (item.IsThrowing())
421 continue;
422
423 if (!item.IsTwoHanded() && !item.IsOneHanded() && !item.IsPolearm())
424 continue;
425
426 var hasAssignedHorse =
427 !assignment.GetEquipmentFromSlot(EquipmentIndex.Horse).IsEmpty &&
428 assignment.GetEquipmentFromSlot(EquipmentIndex.Horse).Item != null;
429
430 if (hasAssignedHorse && !item.IsSuitableForMount())
431 continue;
432
433 candidates.Add(element);
434 }
435
436 if (candidates.Count == 0)
437 return null;
438
439 var randomIndex = MBRandom.RandomInt(candidates.Count);
440 var selected = candidates[randomIndex];
441
442 Global.Log($"(random) weapon {selected.Item.StringId} selected for {_party.Name}", Colors.Green, Level.Debug);
443 return selected;

445 private void AssignEquipmentType(ItemTypeEnum type) {
446 var armours = _equipmentToAssign
447 .Where(kv => kv.Key is { IsEmpty: false, Item: not null } &&
448 kv.Value > 0 &&
449 kv.Key.Item.ItemType == type)
450 .Select(kv => new KeyValuePair<EquipmentElement, int>(kv.Key, kv.Value))
451 .ToArray();
452
453 var settings = ModSettings.Instance;
454 var preferDefaultEquipmentThenClosest = settings?.PreferDefaultEquipmentThenClosest ?? true;
455
456 var slotIndex = Global.GetEquipmentIndexByItemType(type);
457 if (slotIndex == null) { return; }
458
459 for (var assignmentIndex = 0; assignmentIndex < Assignments.Count; assignmentIndex++)
460 {
461 var assignment = Assignments[assignmentIndex];
462 var referenceElement = assignment.ReferenceEquipment.GetEquipmentFromSlot(slotIndex.Value);
463 var referenceItem = referenceElement.Item;
464
465 // Loyal ON: first try exact vanilla equipment.
466 if (preferDefaultEquipmentThenClosest && referenceItem != null)
467 {
468 for (var i = 0; i < armours.Length; i++)
469 {
470 var candidate = armours[i];
471 if (candidate.Value <= 0) { continue; }
472 if (candidate.Key.Item != referenceItem) { continue; }
473
474 assignment.SetEquipment(slotIndex.Value, candidate.Key);
475 armours[i] = new KeyValuePair<EquipmentElement, int>(candidate.Key, candidate.Value - 1);
476 ConsumeEquipmentToAssign(candidate.Key);
477 break;
478 }
479 }
480
481 if (!assignment.Equipment.GetEquipmentFromSlot(slotIndex.Value).IsEmpty) { continue; }
482
483 var maxAllowedTier = GetMaxAllowedTier(referenceItem);
484 var desiredCulture = assignment.Character.Culture;
485
486 // Most-compatible selection (Effectiveness proximity), still capped by +2 tier.
487 if (referenceItem != null)
488 {
489 var targetEffectiveness = referenceItem.Effectiveness;
490
491 var bestCultureIndex = -1;
492 var bestCultureDiff = float.MaxValue;
493 var bestCultureEffectiveness = float.MinValue;
494
495 var bestAnyIndex = -1;
496 var bestAnyDiff = float.MaxValue;
497 var bestAnyEffectiveness = float.MinValue;
498
499 for (var i = 0; i < armours.Length; i++)
500 {
501 var candidate = armours[i];
502 if (candidate.Value <= 0) { continue; }
503
504 var candidateItem = candidate.Key.Item;
505 if (candidateItem == null) { continue; }
506
507 if ((int)candidateItem.Tier > maxAllowedTier) { continue; }
508
509 var diff = Math.Abs(candidateItem.Effectiveness - targetEffectiveness);
510 var effectiveness = candidateItem.Effectiveness;
511
512 // ANY culture
513 if (diff < bestAnyDiff ||
514 diff == bestAnyDiff && effectiveness > bestAnyEffectiveness)
515 {
516 bestAnyIndex = i;
517 bestAnyDiff = diff;
518 bestAnyEffectiveness = effectiveness;
519 }
520
521 // CULTURE-compatible
522 if (!IsCultureCompatible(candidateItem, desiredCulture)) { continue; }
523
524 if (diff < bestCultureDiff ||
525 diff == bestCultureDiff && effectiveness > bestCultureEffectiveness)
526 {
527 bestCultureIndex = i;
528 bestCultureDiff = diff;
529 bestCultureEffectiveness = effectiveness;
530 }
531 }
532
533 var chosenIndex = bestAnyIndex;
534
535 if (preferDefaultEquipmentThenClosest && bestCultureIndex >= 0)
536 {
537 chosenIndex = bestCultureIndex;
538
539 if (bestAnyIndex >= 0)
540 {
541 // diff smaller = more compatible
542 var anyIsMoreThanTwiceAsCompatible = bestCultureDiff > (bestAnyDiff * CULTURE_OVERRIDE_COMPATIBILITY_MULTIPLIER);
543 if (anyIsMoreThanTwiceAsCompatible)
544 {
545 chosenIndex = bestAnyIndex;
546 }
547 }
548 }
549
550
551 if (chosenIndex >= 0)
552 {
553 var chosen = armours[chosenIndex];
554 assignment.SetEquipment(slotIndex.Value, chosen.Key);
555 armours[chosenIndex] = new KeyValuePair<EquipmentElement, int>(chosen.Key, chosen.Value - 1);
556 ConsumeEquipmentToAssign(chosen.Key);
557 }
558
559 continue;
560 }
561
562 // No reference item:
563 {
564 var bestCultureIndex = -1;
565 var bestCultureEffectiveness = float.MinValue;
566
567 var bestAnyIndex = -1;
568 var bestAnyEffectiveness = float.MinValue;
569
570 for (var i = 0; i < armours.Length; i++)
571 {
572 var candidate = armours[i];
573 if (candidate.Value <= 0) { continue; }
574
575 var candidateItem = candidate.Key.Item;
576 if (candidateItem == null) { continue; }
577
578 if ((int)candidateItem.Tier > maxAllowedTier) { continue; }
579
580 var effectiveness = candidateItem.Effectiveness;
581
582 // ANY
583 if (effectiveness > bestAnyEffectiveness)
584 {
585 bestAnyIndex = i;
586 bestAnyEffectiveness = effectiveness;
587 }
588
589 // CULTURE-compatible
590 if (!IsCultureCompatible(candidateItem, desiredCulture)) { continue; }
591
592 if (effectiveness > bestCultureEffectiveness)
593 {
594 bestCultureIndex = i;
595 bestCultureEffectiveness = effectiveness;
596 }
597 }
598
599 var chosenIndex =
600 preferDefaultEquipmentThenClosest
601 ? (bestCultureIndex >= 0 ? bestCultureIndex : bestAnyIndex)
602 : bestAnyIndex;
603
604 if (chosenIndex >= 0)
605 {
606 var chosen = armours[chosenIndex];
607 assignment.SetEquipment(slotIndex.Value, chosen.Key);
608 armours[chosenIndex] = new KeyValuePair<EquipmentElement, int>(chosen.Key, chosen.Value - 1);
609 ConsumeEquipmentToAssign(chosen.Key);
610 }
611 }
612 }

633 private void AssignWeaponByWeaponClassBySlot(EquipmentIndex slot, Assignment assignment, bool mounted, bool strict) {

635
636 var settings = ModSettings.Instance;
637 var preferVanillaThenClosest = settings?.PreferDefaultEquipmentThenClosest ?? true;
638
639 var referenceWeapon = assignment.ReferenceEquipment.GetEquipmentFromSlot(slot);
640 var currentWeapon = assignment.GetEquipmentFromSlot(slot);
641
642 if (currentWeapon is { IsEmpty: false, Item: not null } || referenceWeapon.IsEmpty || referenceWeapon.Item == null)
643 return;
644
645 var maxAllowedTier = GetMaxAllowedTier(referenceWeapon.Item);
646
647 // Loyal ON: exact vanilla weapon first (if available)
648 if (preferVanillaThenClosest &&
649 (int)referenceWeapon.Item.Tier <= maxAllowedTier &&
650 TryConsumeExactItem(referenceWeapon.Item, out var exactElement))
651 {
652 assignment.SetEquipment(slot, exactElement);
653 Global.Log($"weapon (vanilla exact) {exactElement.Item.StringId} assigned to {assignment.Character.StringId}#{assignment.Index} for {_party.Name}", Colors.Green, Level.Debug);
654 return;
655 }
656
657 var desiredCulture = assignment.Character.Culture;
658 var targetEffectiveness = referenceWeapon.Item.Effectiveness;
659
660 EquipmentElement? bestAnyWeapon = null;
661 var bestAnyDiff = float.MaxValue;
662 var bestAnyEffectiveness = float.MinValue;
663
664 EquipmentElement? bestCultureWeapon = null;
665 var bestCultureDiff = float.MaxValue;
666 var bestCultureEffectiveness = float.MinValue;
667
668 foreach (var kv in _equipmentToAssign)
669 {
670 if (kv.Value <= 0)
671 continue;
672
673 var element = kv.Key;
674 if (element.IsEmpty || element.Item == null)
675 continue;
676

841 private bool IsWeaponSuitable(EquipmentElement equipment, ItemObject referenceWeapon, Assignment assignment, bool strict) {
842 if (equipment.IsEmpty ||
843 equipment.Item == null ||
844 !equipment.Item.HasWeaponComponent ||
845 _equipmentToAssign[equipment] <= 0 ||
846 !equipment.Item.IsSuitableForCharacter(assignment.Character) ||
847 !Global.HaveSameWeaponClass(Global.GetWeaponClass(equipment.Item), Global.GetWeaponClass(referenceWeapon))) { return false; }
848
849 var isSuitableForMount = equipment.Item.IsSuitableForMount();
850 var isCouchable = equipment.Item.IsCouchable();
851
852 if (strict) {
853 if (!Global.FullySameWeaponClass(equipment.Item, referenceWeapon)) { return false; }
854
855 if (assignment.IsMounted) {
856 // 严格模式下骑马：必须适合骑乘；如果是长杆武器，则必须可进行骑枪冲刺
857 return isSuitableForMount && (!equipment.Item.IsPolearm() || isCouchable);
858 }
859
860 // 严格模式下非骑马：不可选择可进行骑枪冲刺的武器
861 return !isCouchable;
862 }
863
864 // 非严格模式下：骑马的不能选择不适合骑乘的武器，非骑马的可以选择任意武器
865 return !assignment.IsMounted || isSuitableForMount;
866 }
