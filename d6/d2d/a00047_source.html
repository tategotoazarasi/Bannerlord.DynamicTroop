53 _mission = mission;
54 _party = party;
55 _itemRoster = null;
56 _equipmentToAssign = new ConcurrentDictionary<EquipmentElement, int>(new EquipmentElementComparer());
57
58 foreach (var kv in objectToAssign) {
59 if (kv.Key != null) {
60 EquipmentElement element = new(kv.Key);
61
62 if (_equipmentToAssign.TryGetValue(element, out var existingCount)) { _equipmentToAssign[element] = existingCount + kv.Value; }
63 else { _ = _equipmentToAssign.TryAdd(element, kv.Value); }

68 public void RunAsync() {
69 foreach (var troop in _party.MemberRoster.GetTroopRoster()) {
70 for (var i = 0; i < troop.Number - troop.WoundedNumber; i++) {
71 if (troop.Character is CharacterObject characterObject && !characterObject.IsHero) { Assignments.Add(new Assignment(characterObject)); }
72 }
73 }
74
75 Assignments.Sort((x, y) => y.CompareTo(x));
76 if (_itemRoster != null) {
77 foreach (var kv in _itemRoster) {
78 if (kv is not { IsEmpty: false, EquipmentElement.IsEmpty: false }) { continue; }
79
80 // 尝试获取已存在的数量
81 if (!_equipmentToAssign.TryGetValue(kv.EquipmentElement, out var existingAmount)) {
82 // 如果键不存在，添加新的键值对
83 _ = _equipmentToAssign.TryAdd(kv.EquipmentElement, kv.Amount);
84 }
85 else {
86 // 如果键已存在，更新数量
87 _equipmentToAssign[kv.EquipmentElement] = existingAmount + kv.Amount;
88 }
89 }
90 }
91
92 DoAssignAsync();
93 }

112 private static void PopulateDictionaries(IEnumerable<KeyValuePair<EquipmentElement, int>> equipment,
113 IDictionary<int, List<(EquipmentElement Key, int Cnt)>> horsesDict,
114 IDictionary<int, List<(EquipmentElement Key, int Cnt)>> harnessDict,
115 ICollection<(EquipmentElement Key, int Cnt)> saddles) {
116 foreach (var kvp in equipment) {
117 switch (kvp.Key.Item) {
118 case { HasHorseComponent: true, ItemType: ItemTypeEnum.Horse }:
119 AddToDict(horsesDict, kvp.Key, kvp.Value);
120 break;
121
122 case { HasArmorComponent: true, ItemType: ItemTypeEnum.HorseHarness }:
123 AddToDict(harnessDict, kvp.Key, kvp.Value);
124 break;

126 case {
127 HasArmorComponent : false,
128 HasSaddleComponent: true,
129 ItemType : ItemTypeEnum.HorseHarness
130 }:
131 saddles.Add((kvp.Key, kvp.Value));
132 break;
133 }
134 }

163 private void GenerateHorseAndHarnessPairs(Dictionary<int, List<(EquipmentElement Key, int Cnt)>> horsesDict, IDictionary<int, List<(EquipmentElement Key, int Cnt)>> harnessDict, IList<(EquipmentElement Key, int Cnt)> saddles) {
164 foreach (var kvp in horsesDict) {
165 var familyType = kvp.Key;
166 var horses = kvp.Value.ToArrayQ();
167 var harnesses = harnessDict.TryGetValue(familyType, out var harnessesList) ? harnessesList.ToArrayQ() : Array.Empty<(EquipmentElement Key, int Cnt)>();
168
169 int horseIndex = 0, harnessIndex = 0;
170 while (horseIndex < horses.Length) {
171 (var horseItem, var horseCnt) = horses[horseIndex];
172 var harnessItem = harnessIndex < harnesses.Length ? harnesses[harnessIndex].Key : new EquipmentElement(null);
173 var harnessCnt = harnessIndex < harnesses.Length ? harnesses[harnessIndex].Cnt : 0;
174
175 if (horseCnt > 0) {
176 HorseAndHarness hah = new(horseItem, harnessItem);
177 _horseAndHarnesses.Add(hah);
178 horses[horseIndex] = (horseItem, --horseCnt);
179
180 if (harnessCnt > 0) { harnesses[harnessIndex] = (harnessItem, --harnessCnt); }
181 }
182
183 if (horseCnt == 0) { horseIndex++; }
184
185 if (harnessCnt == 0 && harnessIndex < harnesses.Length) { harnessIndex++; }
186 }
187 }
188
189 AssignSaddlesToHorseAndHarnesses(_horseAndHarnesses, saddles);
190 _horseAndHarnesses.Sort((x, y) => y.CompareTo(x));
191 }

288
289 foreach (var kv in _equipmentToAssign) {
290 if (kv.Key is { IsEmpty: false, Item: not null } &&
291 kv.Value > 0 &&

294

298 candidates.Sort(static (a, b) => {

314 currentIndex++;

405 var candidates = new List<EquipmentElement>();
406
407 foreach (var kv in _equipmentToAssign) {
408 var element = kv.Key;
409 if (kv.Value <= 0 || element.IsEmpty || element.Item == null)
410 continue;
411
412 var item = element.Item;
413 if (!item.HasWeaponComponent)
414 continue;
415
416 if (!item.IsSuitableForCharacter(assignment.Character))
417 continue;
418
419 if (item.IsThrowing())
420 continue;

429 if (hasAssignedHorse && !item.IsSuitableForMount())
430 continue;

432 candidates.Add(element);

434
435 if (candidates.Count == 0)
436 return null;

438 var randomIndex = MBRandom.RandomInt(candidates.Count);

441 Global.Log($"(random) weapon {selected.Item.StringId} selected for {_party.Name}", Colors.Green, Level.Debug);
442 return selected;

445 private void AssignEquipmentType(ItemTypeEnum type) {
446 var armours = _equipmentToAssign
447 .Where(kv => kv.Key is { IsEmpty: false, Item: not null } &&
448 kv.Value > 0 &&
449 kv.Key.Item.ItemType == type)
450 .Select(kv => new KeyValuePair<EquipmentElement, int>(kv.Key, kv.Value))
451 .ToArray();
452
453 Array.Sort(armours,
454 static (a, b) => {
455 var tierCompare = b.Key.Item.Tier.CompareTo(a.Key.Item.Tier);
456 return tierCompare != 0 ? tierCompare : b.Key.Item.Value.CompareTo(a.Key.Item.Value);
457 });
458
459 var settings = ModSettings.Instance;
460 var preferDefaultEquipmentThenClosest = settings?.PreferDefaultEquipmentThenClosest ?? true;
461
462 var slotIndex = Global.GetEquipmentIndexByItemType(type);
463 if (slotIndex == null) { return; }
464
465 for (var assignmentIndex = 0; assignmentIndex < Assignments.Count; assignmentIndex++) {
466 var assignment = Assignments[assignmentIndex];
467 var referenceElement = assignment.ReferenceEquipment.GetEquipmentFromSlot(slotIndex.Value);
468 var referenceItem = referenceElement.Item;
469
470 // Loyal Equipments: first try exact vanilla equipment.
471 if (preferDefaultEquipmentThenClosest && referenceItem != null) {
472 for (var i = 0; i < armours.Length; i++) {
473 var candidate = armours[i];
474 if (candidate.Value <= 0) { continue; }
475
476 if (candidate.Key.Item != referenceItem) { continue; }
477
478 assignment.SetEquipment(slotIndex.Value, candidate.Key);
479 armours[i] = new KeyValuePair<EquipmentElement, int>(candidate.Key, candidate.Value - 1);
480 ConsumeEquipmentToAssign(candidate.Key);
481 break;
482 }
483 }
484
485 if (!assignment.Equipment.GetEquipmentFromSlot(slotIndex.Value).IsEmpty) { continue; }
486
487 var maxAllowedTier = GetMaxAllowedTier(referenceItem);
488
489 // Loyal Equipments: pick the closest-by-tier item (still capped by +2 tier).
490 if (preferDefaultEquipmentThenClosest && referenceItem != null) {
491 var referenceTier = (int)referenceItem.Tier;
492
493 var bestIndex = -1;
494 var bestTierDiff = int.MaxValue;
495 var bestTier = int.MinValue;
496 var bestValue = int.MinValue;
497
498 for (var i = 0; i < armours.Length; i++) {
499 var candidate = armours[i];
500 if (candidate.Value <= 0) { continue; }
501
502 var candidateItem = candidate.Key.Item;
503 if (candidateItem == null) { continue; }
504
505 var candidateTier = (int)candidateItem.Tier;
506 if (candidateTier > maxAllowedTier) { continue; }
507
508 var tierDiff = Math.Abs(candidateTier - referenceTier);
509
510 if (tierDiff < bestTierDiff || tierDiff == bestTierDiff && candidateTier > bestTier || tierDiff == bestTierDiff && candidateTier == bestTier && candidateItem.Value > bestValue) {
511 bestIndex = i;
512 bestTierDiff = tierDiff;
513 bestTier = candidateTier;
514 bestValue = candidateItem.Value;
515 }
516 }
517
518 if (bestIndex >= 0) {
519 var chosen = armours[bestIndex];
520 assignment.SetEquipment(slotIndex.Value, chosen.Key);
521 armours[bestIndex] = new KeyValuePair<EquipmentElement, int>(chosen.Key, chosen.Value - 1);
522 ConsumeEquipmentToAssign(chosen.Key);
523 }
524
525 continue;
526 }
527
528 // Loyal OFF: pick the best available (already sorted best,worst), still capped by +2 tier.
529 for (var i = 0; i < armours.Length; i++) {
530 var candidate = armours[i];
531 if (candidate.Value <= 0) { continue; }
532
533 var candidateItem = candidate.Key.Item;
534 if (candidateItem == null) { continue; }
535
536 if ((int)candidateItem.Tier > maxAllowedTier) { continue; }
537
538 assignment.SetEquipment(slotIndex.Value, candidate.Key);
539 armours[i] = new KeyValuePair<EquipmentElement, int>(candidate.Key, candidate.Value - 1);
540 ConsumeEquipmentToAssign(candidate.Key);
541 break;
542 }
543 }
544 }

564 Global.Log($"AssignWeaponByWeaponClassBySlot slot={slot} character={assignment.Character.StringId}#{assignment.Index} mounted={mounted} strict={strict} for {_party.Name}", Colors.Green, Level.Debug);
565
566 var settings = ModSettings.Instance;
567 var preferVanillaThenClosest = settings?.PreferDefaultEquipmentThenClosest ?? true;
568
569 var referenceWeapon = assignment.ReferenceEquipment.GetEquipmentFromSlot(slot);
570 var currentWeapon = assignment.GetEquipmentFromSlot(slot);
571
572 if (currentWeapon is { IsEmpty: false, Item: not null } || referenceWeapon.IsEmpty || referenceWeapon.Item == null)
573 return;
574
575 var maxAllowedTier = GetMaxAllowedTier(referenceWeapon.Item);
576
577 // Loyal Equipments ON: exact vanilla weapon first (if available)
578 if (preferVanillaThenClosest &&
579 (int)referenceWeapon.Item.Tier <= maxAllowedTier &&
580 TryConsumeExactItem(referenceWeapon.Item, out var exactElement)) {
581 assignment.SetEquipment(slot, exactElement);
582 Global.Log($"weapon (vanilla exact) {exactElement.Item.StringId} assigned to {assignment.Character.StringId}#{assignment.Index} for {_party.Name}", Colors.Green, Level.Debug);
583 return;
584 }
585
586 EquipmentElement? bestWeapon = null;
587
588 var referenceTier = (int)referenceWeapon.Item.Tier;
589 var bestTierDiff = int.MaxValue;
590 var bestTier = int.MinValue;
591 var bestValue = int.MinValue;
592
593 foreach (var kv in _equipmentToAssign) {
594 if (kv.Value <= 0)
595 continue;
596
597 var element = kv.Key;
598 if (element.IsEmpty || element.Item == null)
599 continue;
600
601 var item = element.Item;
602
603 if ((int)item.Tier > maxAllowedTier)
604 continue;
605
606 if (!IsWeaponSuitable(element, referenceWeapon.Item, assignment, strict))
607 continue;
608
609 var itemTier = (int)item.Tier;
610 var tierDiff = Math.Abs(itemTier - referenceTier);
611
612 if (tierDiff < bestTierDiff || tierDiff == bestTierDiff && itemTier > bestTier || tierDiff == bestTierDiff && itemTier == bestTier && item.Value > bestValue) {
613 bestTierDiff = tierDiff;
614 bestTier = itemTier;
615 bestValue = item.Value;
616 bestWeapon = element;
617 }
618 }
619
620
621 AssignWeaponIfAvailable(slot, assignment, bestWeapon);
622 }

636 Global.Log($"AssignWeaponByItemEnumTypeBySlot slot={slot} character={assignment.Character.StringId}#{assignment.Index} mounted={mounted} strict={strict} for {_party.Name}", Colors.Green, Level.Debug);
637
638 var settings = ModSettings.Instance;
639 var preferVanillaThenClosest = settings?.PreferDefaultEquipmentThenClosest ?? true;
640
641 var referenceWeapon = assignment.ReferenceEquipment.GetEquipmentFromSlot(slot);
642 var currentWeapon = assignment.GetEquipmentFromSlot(slot);
643
644 if (currentWeapon is { IsEmpty: false, Item: not null } || referenceWeapon.IsEmpty || referenceWeapon.Item == null)
645 return;
646
647 var maxAllowedTier = GetMaxAllowedTier(referenceWeapon.Item);
648
649 // Loyal Equipments ON: exact vanilla weapon first (if available)
650 if (preferVanillaThenClosest &&
651 (int)referenceWeapon.Item.Tier <= maxAllowedTier &&
652 TryConsumeExactItem(referenceWeapon.Item, out var exactElement)) {
653 assignment.SetEquipment(slot, exactElement);
654 Global.Log($"weapon (vanilla exact) {exactElement.Item.StringId} assigned to {assignment.Character.StringId}#{assignment.Index} for {_party.Name}", Colors.Green, Level.Debug);
655 return;
656 }
657
658 EquipmentElement? bestWeapon = null;
659
660 var referenceTier = (int)referenceWeapon.Item.Tier;
661 var bestTierDiff = int.MaxValue;
662 var bestTier = int.MinValue;
663 var bestValue = int.MinValue;
664
665 foreach (var kv in _equipmentToAssign) {
666 if (kv.Value <= 0)
667 continue;
668
669 var element = kv.Key;
670 if (element.IsEmpty || element.Item == null)
671 continue;
672
673 var item = element.Item;
674
675 if ((int)item.Tier > maxAllowedTier)
676 continue;
677
678 if (!IsWeaponSuitable(element, referenceWeapon.Item, assignment, strict))
679 continue;
680
681 var itemTier = (int)item.Tier;
682 var tierDiff = Math.Abs(itemTier - referenceTier);
683
684 if (tierDiff < bestTierDiff || tierDiff == bestTierDiff && itemTier > bestTier || tierDiff == bestTierDiff && itemTier == bestTier && item.Value > bestValue) {
685 bestTierDiff = tierDiff;
686 bestTier = itemTier;
687 bestValue = item.Value;
688 bestWeapon = element;
689 }
690 }
691
692
693 AssignWeaponIfAvailable(slot, assignment, bestWeapon);
694 }

696 private bool IsWeaponSuitable(EquipmentElement equipment, ItemObject referenceWeapon, Assignment assignment, bool strict) {
697 if (equipment.IsEmpty ||
698 equipment.Item == null ||
699 !equipment.Item.HasWeaponComponent ||
700 _equipmentToAssign[equipment] <= 0 ||
701 !equipment.Item.IsSuitableForCharacter(assignment.Character) ||
702 !Global.HaveSameWeaponClass(Global.GetWeaponClass(equipment.Item), Global.GetWeaponClass(referenceWeapon))) { return false; }
703
704 var isSuitableForMount = equipment.Item.IsSuitableForMount();
705 var isCouchable = equipment.Item.IsCouchable();
706
707 if (strict) {
708 if (!Global.FullySameWeaponClass(equipment.Item, referenceWeapon)) { return false; }
709
710 if (assignment.IsMounted) {
711 // 严格模式下骑马：必须适合骑乘；如果是长杆武器，则必须可进行骑枪冲刺
712 return isSuitableForMount && (!equipment.Item.IsPolearm() || isCouchable);
713 }
714
715 // 严格模式下非骑马：不可选择可进行骑枪冲刺的武器
716 return !isCouchable;
717 }
718
719 // 非严格模式下：骑马的不能选择不适合骑乘的武器，非骑马的可以选择任意武器
720 return !assignment.IsMounted || isSuitableForMount;
721 }

859 private void MarkUnderEquippedAssignmentsIfEnabled() {
860 var settings = ModSettings.Instance;
861 if (!(settings?.Underequipped ?? true)) {
862 for (var i = 0; i < Assignments.Count; i++) {
863 var assignment = Assignments[i];
864 assignment.IsUnderEquipped = false;
865 assignment.UnderEquippedMoralePenalty = 0f;
866 }
867
868 return;
869 }
870
871 for (var i = 0; i < Assignments.Count; i++) {
872 var assignment = Assignments[i];
873
874 var referenceScore = CalculateEquipmentTierScoreWithoutMountEquipment(assignment.ReferenceEquipment);
875 var assignedScore = CalculateEquipmentTierScoreWithoutMountEquipment(assignment.Equipment);
876
877 if (assignedScore >= referenceScore) {
878 assignment.IsUnderEquipped = false;
879 assignment.UnderEquippedMoralePenalty = 0f;
880 continue;
881 }
882
883 var missingTierTotal = referenceScore - assignedScore;
884 var moralePenalty = missingTierTotal * UNDER_EQUIPPED_MORALE_PENALTY_PER_MISSING_TIER;
885
886 assignment.IsUnderEquipped = true;
887 assignment.UnderEquippedMoralePenalty = Math.Min(UNDER_EQUIPPED_MORALE_PENALTY_CAP, moralePenalty);
888 }
889 }
