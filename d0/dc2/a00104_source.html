55 if (MapEvent.PlayerMapEvent == null || MapEvent.PlayerMapEvent.IsRaid) { return; }

59 mainPartyDistributor.RunAsync();
60 Distributors[mainParty.Id] = mainPartyDistributor;

62 PartyBattleSides[mainParty.Id] = PartyBase.MainParty.Side;

64 foreach (var partyBase in MapEvent.PlayerMapEvent.InvolvedParties) {

71 distributor.RunAsync();

73
74 PartyBattleSides[party.Id] = partyBase.Side;

100 public override void OnAgentRemoved(Agent affectedAgent, Agent affectorAgent, AgentState agentState, KillingBlow blow) {
101 if (Mission is not { PlayerTeam.IsValid: true }) {
102 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
103 return;
104 }
105
106
107 if (!affectedAgent.IsValid() || affectedAgent.Character is not { IsHero: false }) {
108 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
109 return;
110 }
111
112 if (_processedAgents.Contains(affectedAgent)) {
113 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
114 return;
115 }
116
117 // potential loot for the opposing side.
118 if (agentState is AgentState.Killed or AgentState.Unconscious) {
119 _ = _processedAgents.Add(affectedAgent);
120
121 var affectedPartyId = Global.GetAgentParty(affectedAgent.Origin)?.Id;
122 if (!affectedPartyId.HasValue) {
123 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
124 return;
125 }
126
127 if (!_partyBattleRecords.ContainsKey(affectedPartyId.Value))
128 _partyBattleRecords.TryAdd(affectedPartyId.Value, new PartyBattleRecord());
129
130 var affectedBattleRecord = _partyBattleRecords[affectedPartyId.Value];
131
132 PartyBattleRecord? affectorBattleRecord = null;
133 var affectorPartyId = affectorAgent.IsValid() ? Global.GetAgentParty(affectorAgent.Origin)?.Id : null;
134 if (affectorPartyId.HasValue) {
135 if (!_partyBattleRecords.ContainsKey(affectorPartyId.Value))
136 _partyBattleRecords.TryAdd(affectorPartyId.Value, new PartyBattleRecord());
137
138 affectorBattleRecord = _partyBattleRecords[affectorPartyId.Value];
139 }
140
141 var hitBodyPart = blow.VictimBodyPart;
142 var armors = affectedAgent.GetAgentArmors();
143 var hitArmor = ArmorSelector.GetRandomArmorByBodyPart(armors, hitBodyPart);
144
145 ProcessAgentEquipmentRespectingTemporarySlots(
146 affectedAgent,
147 item => {
148 if (item == null || !ItemBlackList.Test(item))
149 return;
150
151 if (hitArmor != null && item.StringId == hitArmor.StringId)
152 return;
153
154 var dropRate = SubModule.Settings?.DropRate ?? 1f;
155
156 affectedBattleRecord.AddItemToRecover(item);
157
158 if (affectorBattleRecord != null && _random.NextFloat() <= dropRate) {
159 affectorBattleRecord.AddLootedItem(item);
160 Global.Log($"{item.StringId} looted", Colors.Green, Level.Debug);
161 }
162 });
163 }
164
165 //Routed (alive, removed mid-battle), make sure their equipment is still returned to their own armory
166 else if (!_isMissionEnded && agentState is AgentState.Routed or AgentState.Deleted) {
167 _ = _processedAgents.Add(affectedAgent);
168 var affectedPartyId = Global.GetAgentParty(affectedAgent.Origin)?.Id;
169 if (!affectedPartyId.HasValue) {
170 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
171 return;
172 }
173
174 if (!_partyBattleRecords.ContainsKey(affectedPartyId.Value))
175 _partyBattleRecords.TryAdd(affectedPartyId.Value, new PartyBattleRecord());
176
177 var affectedBattleRecord = _partyBattleRecords[affectedPartyId.Value];
178 ProcessAgentEquipmentRespectingTemporarySlots(affectedAgent, item => { affectedBattleRecord.AddItemToRecover(item); });
179 Global.Log(
180 $"[DTES] Killed/Unconscious processed for {affectedAgent.Character?.Name} | party={affectedPartyId.Value} | recordRecoverNow={affectedBattleRecord.ItemsToRecoverCount}",
181 Colors.Green,
182 Level.Debug);
183 }
184
185 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
186 }

225 private void OnMissionEnded(MissionResult? missionResult) {
226 Global.Log("OnMissionEnded() called", Colors.Green, Level.Debug);
227 if (_isMissionEnded) return;
228
229 _isMissionEnded = true;
230
231 var playerVictory = missionResult?.PlayerVictory ?? false;
232 var playerDefeat = missionResult?.PlayerDefeated ?? false;
233 var unresolvedBattle = missionResult == null || !missionResult.BattleResolved;
234
235 // 确定玩家方
236 var playerPartyId = Campaign.Current.MainParty.Id;
237
238 foreach (var kvPartyBattleSides in PartyBattleSides) {
239 var isPlayerParty = kvPartyBattleSides.Key == playerPartyId;
240 var isVictorious = !unresolvedBattle && (playerVictory && kvPartyBattleSides.Value == Mission.PlayerTeam.Side || playerDefeat && kvPartyBattleSides.Value != Mission.PlayerTeam.Side);
241 var isDefeated = !unresolvedBattle && (playerDefeat && kvPartyBattleSides.Value == Mission.PlayerTeam.Side || playerVictory && kvPartyBattleSides.Value != Mission.PlayerTeam.Side);
242
243 HandlePartyItems(kvPartyBattleSides.Key, isPlayerParty, isVictorious, isDefeated);
244 }
245
246 // 回收场上士兵的装备
247 foreach (var kvPartyBattleSides in PartyBattleSides) {
248 IEnumerable<Agent> partyAgents = Mission.Agents.WhereQ(agent =>
249 agent.IsValid() &&
250 agent.IsActive() &&
251 !agent.IsHero &&
252 !_processedAgents.Contains(agent) &&
253 Global.GetAgentParty(agent.Origin)?.Id == kvPartyBattleSides.Key);
254 ReturnEquipmentFromAgents(kvPartyBattleSides.Key, partyAgents);
255 }
256 }

265 private void HandlePartyItems(MBGUID partyId, bool isPlayerParty, bool isVictorious, bool isDefeated) {
266 // 检查是否存在对应的battleRecord
267 if (!_partyBattleRecords.TryGetValue(partyId, out var battleRecord)) {
268 Global.Warn($"No battle record found for party {partyId}");
269 return;
270 }

272 if (isVictorious) {
273 ReturnItemsToDestination(partyId, battleRecord.ItemsToRecover, isPlayerParty);

278 }
279 }
280 else if (!isDefeated) { ReturnItemsToDestination(partyId, battleRecord.ItemsToRecover, isPlayerParty); }

291 private void ReturnItemsToDestination(MBGUID partyId, ConcurrentDictionary<ItemObject, int> items, bool isPlayerParty) {
292 var totalItemCount = 0; // 用于累计物品总数
293
294 foreach (var item in items) {
295 if (item.Key == null || item.Value <= 0 || !ItemBlackList.Test(item.Key)) continue;
296
297 totalItemCount += item.Value;
298
299 if (isPlayerParty)
300 ArmyArmory.AddItemToArmory(item.Key, item.Value);
301 else if (Distributors.TryGetValue(partyId, out var distributor))
302 distributor.ReturnItem(item.Key, item.Value);
303 }
304
305 if (totalItemCount <= 0) return;
306
307 var partyType = isPlayerParty ? "Player party" : "Party";
308 Global.Log($"{partyType} {partyId} processed a total of {totalItemCount} items", Colors.Green, Level.Debug);
309 }

343 private void ProcessAgentEquipmentRespectingTemporarySlots(Agent agent, Action<ItemObject> processEquipmentItem) {
344 if (_assignmentByAgent.TryGetValue(agent, out var assignment)) {
345 var processedItemCount = 0;
346
347 Global.ProcessAgentEquipment(
348 agent,
349 item => {
350 processedItemCount++;
351 processEquipmentItem(item);
352 },
353 (slot, missionWeapon, spawnElement) => !assignment.IsTemporarySlot(slot, spawnElement.Item));
354
355 if (processedItemCount == 0) {
356 Global.Log($"[DTES] Agent equipment returned 0 items. Fallback to Assignment. Agent={agent.Character?.Name}", Colors.Yellow, Level.Debug);
357 ProcessAssignmentEquipmentFallback(assignment, processEquipmentItem);
358 }
359
360 return;
361 }
362
363 Global.ProcessAgentEquipment(agent, processEquipmentItem);
364 }

366 private static void ProcessAssignmentEquipmentFallback(Assignment assignment, Action<ItemObject> processEquipmentItem) {
367 foreach (var slot in Global.EquipmentSlots) {
368 var element = assignment.Equipment[slot];
369 if (element.IsEmpty || element.Item is null)
370 continue;
371
372 if (element.Item.IsBannerItem || element.Item.ItemType == ItemObject.ItemTypeEnum.Banner)
373 continue;
374
375 if (assignment.IsTemporarySlot(slot, element.Item))
376 continue;
377
378 if (element.Item.ItemType is ItemObject.ItemTypeEnum.Arrows
379 or ItemObject.ItemTypeEnum.Bolts
380 or ItemObject.ItemTypeEnum.Thrown) { continue; }
381
382 processEquipmentItem(element.Item);
383 }
