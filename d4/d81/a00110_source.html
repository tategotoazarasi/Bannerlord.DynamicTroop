121 public static bool Test(ItemObject item) {
122 try {
123 if (Cache.TryGetValue(item, out var cachedResult))
124 return cachedResult;
125
126 if (item == null || item.StringId.IsEmpty() || item.Name == null || item.Name.ToString().IsEmpty())
127 return true;
128
129 var stringId = item.StringId;
130 var name = item.Name.ToString();
131
132 var isAllowed =
133 !StringIds.Contains(stringId) &&
134 !Names.Contains(name) &&
135 !StringIdPatterns.Any(pattern => pattern.IsMatch(stringId)) &&
136 !NamePatterns.Any(pattern => pattern.IsMatch(name));
137
138 Cache[item] = isAllowed;
139 return isAllowed;
140 }
141 catch (Exception e) {
142 Global.Error(e.Message);
143 return false;
144 }
