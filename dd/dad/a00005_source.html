128 public int CompareTo(object? obj) {
129 if (obj == null) return 1;
130
131 if (obj is not Assignment other) throw new ArgumentException("Object is not an Assignment");
132
133 var tierComparison = Character.Tier.CompareTo(other.Character.Tier);
134 if (tierComparison != 0) return tierComparison;
135
136 // TroopType 比较，小的放在前面
137 var thisTroopType = Character.GetTroopType();
138 var otherTroopType = other.Character.GetTroopType();
139 var troopTypeComparison = thisTroopType.CompareTo(otherTroopType);
140 if (troopTypeComparison != 0) return troopTypeComparison;
141
142 // 按 IsRanged 比较，IsRanged 排在 !IsRanged 前面
143 if (!Character.IsRanged && other.Character.IsRanged) return 1;
144
145 if (Character.IsRanged && !other.Character.IsRanged) return -1;
146
147 // 按 IsMounted 比较，!IsMounted 排在 IsMounted 前面
148 if (Character.IsMounted && !other.Character.IsMounted) return 1;
149
150 if (!Character.IsMounted && other.Character.IsMounted) return -1;
151
152 var skillValueComparison = Character.SkillValue().CompareTo(other.Character.SkillValue());
153 if (skillValueComparison != 0) return skillValueComparison;
154
155 var equipmentValueComparison = Character.EquipmentValue().CompareTo(other.Character.EquipmentValue());
156 if (equipmentValueComparison != 0) return equipmentValueComparison;
157
158 var levelComparison = Character.Level.CompareTo(other.Character.Level);
159 if (levelComparison != 0) return levelComparison;
160
161 return 0; // 如果所有条件都相等，则认为两者相等
162 }

204 public void FillEmptySlots() {
205 var isHideoutBattle = Mission.Current?.HasMissionBehavior<HideoutMissionController>() ?? false;
206 var isSiegeBattle = Mission.Current?.HasMissionBehavior<MissionSiegeEnginesLogic>() ?? false;
207
208 foreach (var slot in Global.ArmourSlots) {
209 var referenceEquipment = ReferenceEquipment.GetEquipmentFromSlot(slot);
210 if (GetEquipmentFromSlot(slot) is not { IsEmpty: false, Item: not null }) {
211 var itemType = Helper.EquipmentIndexToItemEnumType(slot);
212 if (!itemType.HasValue) continue;
213
214 ItemObject? item;
215 if (referenceEquipment is { IsEmpty: false, Item: not null }) {
216 List<ItemObject> itemList = new() { referenceEquipment.Item };
217 var itemsByCharacter = Cache.GetItemsByTypeTierAndCulture(itemType.Value, Character.Tier, Character.Culture);
218 if (itemsByCharacter != null) itemList.AddRange(itemsByCharacter);
219
220 if (referenceEquipment.Item.Culture is CultureObject cultureObject) {
221 var itemsByReference = Cache.GetItemsByTypeTierAndCulture(itemType.Value, (int)referenceEquipment.Item.Tier, cultureObject);
222 if (itemsByReference != null) itemList.AddRange(itemsByReference);
223 }
224
225 item = WeightedRandomSelector.SelectItem(itemList, referenceEquipment.Item.Effectiveness);
226 }
227 else {
228 item = Cache.GetItemsByTypeTierAndCulture(itemType.Value, Character.Tier, Character.Culture)?.GetRandomElement();
229 if (item == null) continue;
230 }
231
232 SetEquipment(slot, new EquipmentElement(item));
233 }
234 }
235
236 foreach (var slot in Global.EquipmentSlots) {
237 // hideout and siege harness fix
238 if (isHideoutBattle && (slot == EquipmentIndex.Horse || slot == EquipmentIndex.HorseHarness)) continue;
239 if ((isHideoutBattle || isSiegeBattle) && (slot == EquipmentIndex.Horse || slot == EquipmentIndex.HorseHarness)) continue;
240
241 if (GetEquipmentFromSlot(slot) is not { IsEmpty: false, Item: not null })
