98 public static void Log(string message, Color color, Level level, int skipFrames = 1) {
99 if (SubModule.Settings is { DebugMode: true } && (SubModule.Settings.LogLevel.SelectedValue == Level.All || level >= SubModule.Settings.LogLevel.SelectedValue)) {
100 MessageDisplayService.EnqueueMessage(new InformationMessage(message, color));
101
102 // 使用 log4net 记录日志
103 StackFrame frame = new(skipFrames, true); // 创建 StackFrame 对象，参数 1 表示上一个栈帧
104 var method = frame.GetMethod(); // 获取方法信息
105
106 // 获取文件名而不是完整路径
107 var fileName = Path.GetFileName(frame.GetFileName());
108
109 var lineNumber = frame.GetFileLineNumber(); // 获取行号
110
111 // 构造日志消息
112 var logMessage = $"[{method.DeclaringType?.FullName}.{method.Name}] [{fileName}:{lineNumber}] {message}";
113
114 // 获取 log4net 日志实例
115 //LogManager.GetLogger(method.DeclaringType);
116 var log = LogManager.GetLogger("dtes");
117
118 // 根据指定的日志级别记录日志
119 if (level == Level.Debug)
120 log.Debug(logMessage);
121 else if (level == Level.Info)
122 log.Info(logMessage);
123 else if (level == Level.Warn)
124 log.Warn(logMessage);
125 else if (level == Level.Error)
126 log.Error(logMessage);
127 else if (level == Level.Fatal) log.Fatal(logMessage);
128 }
129 }

165 public static void ProcessAgentEquipment(Agent agent, Action<ItemObject> processEquipmentItem) {
166 if (!agent.IsValid()) return;
167
168 var missionEquipment = agent.Equipment;
169 var spawnEquipment = agent.SpawnEquipment;
170
171 if (missionEquipment == null || spawnEquipment == null) return;
172
173 // 处理武器槽装备
174 foreach (var slot in Assignment.WeaponSlots) {
175 var element = missionEquipment[slot];
176 if (element.IsEmpty || element.Item == null) continue;
177
178 if (IsAmmoAndEmpty(element)) {
179 Log($"Empty Ammo {element.Item.StringId}", Colors.Green, Level.Debug);
180 continue;
181 }
182
183 processEquipmentItem(element.Item);
184 }
185
186 // 处理装甲和马匹槽装备
187 foreach (var slot in ArmourAndHorsesSlots) {
188 var element = spawnEquipment[slot];
189 if (element.IsEmpty || element.Item == null) continue;
190
191 processEquipmentItem(element.Item);
192 }
193 }

230 public static List<ItemObject> CreateRandomCraftedItemsByItemType(ItemObject.ItemTypeEnum? type, BasicCultureObject? culture, int num = 0) {
231 List<ItemObject> items = new();
232 Random random = new();
233 if (type == null || culture == null) return items;
234
235 var templates = CraftingTemplatesByItemType[type.Value];
236 for (var i = 0; i < num; i++) {
237 var randomElement = templates[random.Next() % templates.Length];
238 TextObject textObject = new("{=uZhHh7pm}Crafted {CURR_TEMPLATE_NAME}");
239 _ = textObject.SetTextVariable("CURR_TEMPLATE_NAME", randomElement.TemplateName);
240 Crafting crafting = new(randomElement, culture, textObject);
241 crafting.Init();
242 crafting.Randomize();
243 var hashedCode = crafting.CurrentWeaponDesign.HashedCode;
244 crafting.GetCurrentCraftedItemObject().StringId = hashedCode;
245 var itemObject = MBObjectManager.Instance.GetObject<ItemObject>(hashedCode);
246 if (itemObject == null) {
247 itemObject = MBObjectManager.Instance.RegisterObject(crafting.GetCurrentCraftedItemObject());
248 items.Add(itemObject);
249 }
250 }
251
252 return items;
253 }
