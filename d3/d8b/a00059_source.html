105 public override void OnAgentRemoved(Agent affectedAgent, Agent affectorAgent, AgentState agentState, KillingBlow blow) {
106 if (Mission is not { PlayerTeam.IsValid: true }) {
107 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
108 return;
109 }
110
111
112 if (!affectedAgent.IsValid() || affectedAgent.Character is not { IsHero: false }) {
113 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
114 return;
115 }
116
117 if (_processedAgents.Contains(affectedAgent)) {
118 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
119 return;
120 }
121
122 // potential loot for the opposing side.
123 if (agentState is AgentState.Killed or AgentState.Unconscious) {
124 _ = _processedAgents.Add(affectedAgent);
125
126 var affectedPartyId = Global.GetAgentParty(affectedAgent.Origin)?.Id;
127 if (!affectedPartyId.HasValue) {
128 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
129 return;
130 }
131
132 if (!_partyBattleRecords.ContainsKey(affectedPartyId.Value))
133 _partyBattleRecords.TryAdd(affectedPartyId.Value, new PartyBattleRecord());
134
135 var affectedBattleRecord = _partyBattleRecords[affectedPartyId.Value];
136
137 var hasAffectedSide = PartyBattleSides.TryGetValue(affectedPartyId.Value, out var affectedSide);
138 var affectorPartyId = affectorAgent.IsValid() ? Global.GetAgentParty(affectorAgent.Origin)?.Id : null;
139
140 PartyBattleRecord? affectorBattleRecord = null;
141 if (hasAffectedSide &&
142 affectorPartyId.HasValue &&
143 PartyBattleSides.TryGetValue(affectorPartyId.Value, out var affectorSide) &&
144 affectorSide != affectedSide)
145 affectorBattleRecord = _partyBattleRecords.GetOrAdd(
146 affectorPartyId.Value,
147 _ => new PartyBattleRecord());
148
149 var hitBodyPart = blow.VictimBodyPart;
150 var armors = affectedAgent.GetAgentArmors();
151 var hitArmor = ArmorSelector.GetRandomArmorByBodyPart(armors, hitBodyPart);
152 // Extra armor break: ONLY for enemy casualties, tierbased.
153 // This is an addition to the existing hitArmor destruction logic.
154 ItemObject? extraBrokenArmor = null;
155
156 var isEnemyCasualty = hasAffectedSide && affectedSide != Mission.PlayerTeam.Side;
157
158 if (isEnemyCasualty)
159 {
160 var troopTier = 0;
161 if (affectedAgent.Character is CharacterObject characterObject)
162 troopTier = characterObject.Tier;
163
164 var extraBreakChance = troopTier switch {
165 3 => 0.25f,
166 4 => 0.35f,
167 5 => 0.45f,
168 _ => troopTier >= 6 ? 0.60f : 0f
169 };
170
171 if (extraBreakChance > 0f && _random.NextFloat() <= extraBreakChance)
172 {
173 // Pick an additional armor piece to destroy (excluding hitArmor)
174 var candidates = new List<ItemObject>();
175
176 foreach (var armor in armors)
177 {
178 if (armor == null)
179 continue;
180
181 if (hitArmor != null && armor.StringId == hitArmor.StringId)
182 continue;
183
184 if (armor.ItemType == ItemObject.ItemTypeEnum.HorseHarness)
185 continue;
186
187 candidates.Add(armor);
188 }
189
190 if (candidates.Count > 0)
191 extraBrokenArmor = candidates[_random.Next(candidates.Count)];
192 }
193 }
194
195
196 ProcessAgentEquipmentRespectingTemporarySlots(
197 affectedAgent,
198 item => {
199 if (item == null || !ItemBlackList.Test(item))
200 return;
201
202 if (hitArmor != null && item.StringId == hitArmor.StringId)
203 return;
204
205 if (extraBrokenArmor != null && item.StringId == extraBrokenArmor.StringId)
206 return;
207
208 var dropRate = SubModule.Settings?.DropRate ?? 1f;
209
210 affectedBattleRecord.AddItemToRecover(item);
211
212 if (_random.NextFloat() > dropRate)
213 return;
214
215 if (affectorBattleRecord != null)
216 {
217 affectorBattleRecord.AddLootedItem(item);
218 Global.Log($"{item.StringId} queued for naval casualty loot", Colors.Green, Level.Debug);
219 return;
220 }
221
222 if (!Mission.IsNavalBattle && !Mission.IsNavalRaidBattle)
223 return;
224
225 var casualtyLoot = _unclaimedNavalCasualtyLoot.GetOrAdd(
226 affectedPartyId.Value,
227 _ => new ConcurrentDictionary<ItemObject, int>());
228 casualtyLoot.AddOrUpdate(item, 1, (_, currentCount) => currentCount + 1);
229 Global.Log($"{item.StringId} looted", Colors.Green, Level.Debug);
230 });
231 }
232
233 //Routed (alive, removed mid-battle), make sure their equipment is still returned to their own armory
234 else if (!_isMissionEnded && agentState is AgentState.Routed or AgentState.Deleted) {
235 _ = _processedAgents.Add(affectedAgent);
236 var affectedPartyId = Global.GetAgentParty(affectedAgent.Origin)?.Id;
237 if (!affectedPartyId.HasValue) {
238 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
239 return;
240 }
241
242 if (!_partyBattleRecords.ContainsKey(affectedPartyId.Value))
243 _partyBattleRecords.TryAdd(affectedPartyId.Value, new PartyBattleRecord());
244
245 var affectedBattleRecord = _partyBattleRecords[affectedPartyId.Value];
246 ProcessAgentEquipmentRespectingTemporarySlots(affectedAgent, item => { affectedBattleRecord.AddItemToRecover(item); });
247 Global.Log(
248 $"[DTES] Routed/deleted equipment recorded for {affectedAgent.Character?.Name} | party={affectedPartyId.Value} | recordRecoverNow={affectedBattleRecord.ItemsToRecoverCount}",
249 Colors.Green,
250 Level.Debug);
251 }
252
253 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
254 }

295 private void FinalizeMission(MissionResult? missionResult) {
296 Global.Log("FinalizeMission() called", Colors.Green, Level.Debug);
297 if (_isMissionEnded) return;
298
299 if (!_areDistributorsInitialized) {
300 _isMissionEnded = true;
301 return;
302 }
303
304 var mainParty = Campaign.Current?.MainParty;
305 var playerTeam = Mission.PlayerTeam;
306 if (mainParty == null || playerTeam == null) return;
307
308 _isMissionEnded = true;
309
310 var playerVictory = missionResult?.PlayerVictory ?? false;
311 var playerDefeat = missionResult?.PlayerDefeated ?? false;
312 var unresolvedBattle = missionResult == null || !missionResult.BattleResolved;
313 var playerPartyId = mainParty.Id;
314 var playerSide = playerTeam.Side;
315 var winningSide = missionResult?.BattleState switch {
316 BattleState.AttackerVictory => BattleSideEnum.Attacker,
317 BattleState.DefenderVictory => BattleSideEnum.Defender,
318 _ => BattleSideEnum.None
319 };
320
321 if (winningSide != BattleSideEnum.None &&
322 (Mission.IsNavalBattle || Mission.IsNavalRaidBattle))
323 DistributeNavalCasualtyLoot(winningSide);
324
325 foreach (var partyBattleSide in PartyBattleSides)
326 {
327 var isPlayerParty = partyBattleSide.Key == playerPartyId;
328 var isVictorious = !unresolvedBattle &&
329 (playerVictory && partyBattleSide.Value == playerSide ||
330 playerDefeat && partyBattleSide.Value != playerSide);
331 var isDefeated = !unresolvedBattle &&
332 (playerDefeat && partyBattleSide.Value == playerSide ||
333 playerVictory && partyBattleSide.Value != playerSide);
334
335 HandlePartyItems(partyBattleSide.Key, isPlayerParty, isVictorious, isDefeated);
336 }
337
338 foreach (var partyBattleSide in PartyBattleSides) {
339 IEnumerable<Agent> partyAgents = Mission.Agents.WhereQ(agent =>
340 agent.IsValid() &&
341 agent.IsActive() &&
342 !agent.IsHero &&
343 !_processedAgents.Contains(agent) &&
344 Global.GetAgentParty(agent.Origin)?.Id == partyBattleSide.Key);
345 ReturnEquipmentFromAgents(partyBattleSide.Key, partyAgents, playerPartyId);
346 }

348 private void DistributeNavalCasualtyLoot(BattleSideEnum winningSide) {
349 if (_unclaimedNavalCasualtyLoot.IsEmpty)
350 return;
351
352 var mapEvent = MapEvent.PlayerMapEvent;
353 if (mapEvent == null)
354 return;
355
356 var defeatedSide = winningSide == BattleSideEnum.Attacker
357 ? BattleSideEnum.Defender
358 : BattleSideEnum.Attacker;
359 var winnerParties = mapEvent.PartiesOnSide(winningSide);
360 var defeatedParties = mapEvent.PartiesOnSide(defeatedSide);
361
362 foreach (var defeatedPartyLoot in _unclaimedNavalCasualtyLoot)
363 {
364 MapEventParty? defeatedParty = null;
365 foreach (var candidate in defeatedParties)
366 {
367 if (candidate.Party.MobileParty?.Id != defeatedPartyLoot.Key)
368 continue;
369
370 defeatedParty = candidate;
371 break;
372 }
373
374 if (defeatedParty == null)
375 continue;
376
377 var lootChances = Campaign.Current.Models.BattleRewardModel
378 .GetLootCasualtyChances(winnerParties, defeatedParty.Party);
379
380 foreach (var item in defeatedPartyLoot.Value)
381 {
382 for (var itemIndex = 0; itemIndex < item.Value; itemIndex++)
383 {
384 var recipientParty = NavalCasualtyLootOwner(lootChances, winningSide);
385 if (recipientParty == null)
386 continue;
387
388 var battleRecord = _partyBattleRecords.GetOrAdd(
389 recipientParty.Id,
390 _ => new PartyBattleRecord());
391 battleRecord.AddLootedItem(item.Key);
392 }
393 }
394 }

397 private MobileParty? NavalCasualtyLootOwner(
398 MBReadOnlyList<KeyValuePair<MapEventParty, float>> lootChances,
399 BattleSideEnum winningSide) {
400 var totalChance = 0f;
401 foreach (var lootChance in lootChances)
402 {
403 var mobileParty = lootChance.Key.Party.MobileParty;
404 if (mobileParty != null &&
405 lootChance.Value > 0f &&
406 PartyBattleSides.TryGetValue(mobileParty.Id, out var partySide) &&
407 partySide == winningSide)
408 totalChance += lootChance.Value;
409 }
410
411 if (totalChance <= 0f)
412 return null;
413
414 var roll = _random.NextDouble() * totalChance;
415 var accumulatedChance = 0f;
416 MobileParty? lastEligibleParty = null;
417
418 foreach (var lootChance in lootChances)
419 {
420 var mobileParty = lootChance.Key.Party.MobileParty;
421 if (mobileParty == null ||
422 lootChance.Value <= 0f ||
423 !PartyBattleSides.TryGetValue(mobileParty.Id, out var partySide) ||
424 partySide != winningSide)
425 continue;
426
427 lastEligibleParty = mobileParty;
428 accumulatedChance += lootChance.Value;
429 if (roll < accumulatedChance)
430 return mobileParty;
431 }
432
433 return lastEligibleParty;
434 }

469 private void ReturnItemsToDestination(MBGUID partyId, ConcurrentDictionary<ItemObject, int> items, bool isPlayerParty) {
470 var totalItemCount = 0; // 用于累计物品总数
471
472 foreach (var item in items) {
473 if (item.Key == null || item.Value <= 0 || !ItemBlackList.Test(item.Key)) continue;
474
475 totalItemCount += item.Value;
476
477 if (isPlayerParty)
478 ArmyArmory.AddItemToArmory(item.Key, item.Value);
479 else if (Distributors.TryGetValue(partyId, out var distributor))
480 distributor.ReturnItem(item.Key, item.Value);
481 }
482
483 if (totalItemCount <= 0) return;
484
485 var partyType = isPlayerParty ? "Player party" : "Party";
486 Global.Log($"{partyType} {partyId} processed a total of {totalItemCount} items", Colors.Green, Level.Debug);

544 private static void ProcessAssignmentEquipmentFallback(Assignment assignment, Action<ItemObject> processEquipmentItem) {
545 foreach (var slot in Global.EquipmentSlots) {
546 if (!assignment.CanUseMountEquipment && (slot == EquipmentIndex.Horse || slot == EquipmentIndex.HorseHarness))
547 continue;
548
549 var element = assignment.Equipment[slot];
550 if (element.IsEmpty || element.Item is null)
551 continue;
552
553 if (element.Item.IsBannerItem || element.Item.ItemType == ItemObject.ItemTypeEnum.Banner)
554 continue;
555
556 if (assignment.IsTemporarySlot(slot, element.Item))
557 continue;
558
559 if (element.Item.ItemType is ItemObject.ItemTypeEnum.Arrows
560 or ItemObject.ItemTypeEnum.Bolts
561 or ItemObject.ItemTypeEnum.Thrown) { continue; }
562
563 processEquipmentItem(element.Item);
564 }
