126 private void GarbageCollectEquipments(MobileParty mobileParty) {
127 if (!mobileParty.IsValid() || mobileParty.MemberRoster?.GetTroopRoster() == null) return;
128
129 var memberCnt = mobileParty.MemberRoster.GetTroopRoster()
130 .WhereQ(element => element.Character is { IsHero: false })
131 .SumQ(element => element.Number);
132 foreach (var equipmentAndThreshold in EquipmentAndThresholds) {
133 var partyArmory = PartyArmories.GetValueSafe(mobileParty.Id);
134 if (partyArmory == null || partyArmory.IsEmpty()) return;
135
136 var armorTotalCount = partyArmory.WhereQ(kv => kv.Key.ItemType == equipmentAndThreshold.Key)
137 .SumQ(kv => kv.Value);
138 var surplusCount = armorTotalCount - memberCnt;
139 if (surplusCount <= 0) continue;
140
141 var surplusCountCpy = surplusCount;
142
143 // 创建优先级队列
144 ItemPriorityQueue armorQueue = new(new EquipmentEffectivenessComparer());
145 foreach (var kv in partyArmory.WhereQ(kv => kv.Key.ItemType == equipmentAndThreshold.Key))
146 armorQueue.Enqueue(kv.Key, ((int)kv.Key.Tier, kv.Key.Value));
147
148 // 移除多余的装备
149 while (surplusCount > 0 && armorQueue.Count > 0) {
150 var lowestArmor = armorQueue.Dequeue();
151 var countToRemove = Math.Min(partyArmory[lowestArmor.Key], surplusCount);
152 partyArmory[lowestArmor.Key] -= countToRemove;
153 surplusCount -= countToRemove;
154 if (partyArmory[lowestArmor.Key] == 0) _ = partyArmory.Remove(lowestArmor.Key);
155 }
156
157 Global.Debug($"Garbage collected {surplusCountCpy - surplusCount}x{equipmentAndThreshold.Key} armors from party {mobileParty.Name}");
158 }
159 }

338 Dictionary<ItemObject, int> itemsWithCount = new();

341 foreach (var item in inventory) {

343 if (itemsWithCount.ContainsKey(item.Key))

480 private static Dictionary<MBGUID, Dictionary<ItemObject, int>> ConvertToGuidDict(
481 Dictionary<uint, Dictionary<string, int>> uintDict) {
482 Dictionary<MBGUID, Dictionary<ItemObject, int>> guidDict = new();
483 foreach (var pair in uintDict) {
484 if (!guidDict.TryGetValue(new MBGUID(pair.Key), out var innerDict)) {
485 innerDict = new Dictionary<ItemObject, int>();
486 guidDict.Add(new MBGUID(pair.Key), innerDict);
487 }
488
489 foreach (var innerPair in pair.Value) {
490 var itemObject = MBObjectManager.Instance.GetObject<ItemObject>(innerPair.Key) ??
491 ItemObject.GetCraftedItemObjectFromHashedCode(innerPair.Key);
492 if (itemObject != null && innerPair.Value > 0) {
493 if (innerDict.ContainsKey(itemObject))
494 innerDict[itemObject] += innerPair.Value;
495 else
496 innerDict.Add(itemObject, innerPair.Value);
497 }
498 else { Global.Warn($"cannot get object {innerPair.Key}"); }
499 }
500 }
501
502 return guidDict;
