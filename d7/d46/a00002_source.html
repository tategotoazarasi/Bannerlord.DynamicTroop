98 private static int SellExcessEquipment() {
99 RebuildArmory();
100 var excessValue = 0;
101 var playerParty = MobileParty.MainParty;
102 if (playerParty?.MemberRoster?.GetTroopRoster() == null) return 0;
103
104 var memberCnt = playerParty.MemberRoster.GetTroopRoster()
105 .WhereQ(element => element.Character is { IsHero: false })
106 .SumQ(element => element.Number);
107
108 foreach (var equipmentAndThreshold in EveryoneCampaignBehavior.EquipmentAndThresholds) {
109 var armorTotalCount = Armory.WhereQ(kv => kv.EquipmentElement.Item?.ItemType == equipmentAndThreshold.Key)
110 .SumQ(kv => kv.Amount);
111 var surplusCount = armorTotalCount - equipmentAndThreshold.Value(memberCnt);
112 if (surplusCount <= 0) continue;
113
114 var surplusCountCpy = surplusCount;
115
116 // 创建优先级队列
117 ItemPriorityQueue armorQueue = new(new ArmorElementComparer());
118 foreach (var kv in Armory.WhereQ(kv => kv.EquipmentElement.Item?.ItemType == equipmentAndThreshold.Key))
119 armorQueue.Enqueue(kv.EquipmentElement, kv.EquipmentElement.ItemValue);
120
121 // 移除多余的装备
122 while (surplusCount > 0 && !armorQueue.IsEmpty) {
123 var lowestArmor = armorQueue.Dequeue();
124 var countToRemove = Math.Min(Armory.GetElementNumber(Armory.FindIndexOfElement(lowestArmor.Key)),
125 surplusCount);
126
127 //Global.Debug($"countToRemove={countToRemove}, lowestArmorNumber={Armory.GetItemNumber(lowestArmor.Key)}");
128 _ = Armory.AddToCounts(lowestArmor.Key, -countToRemove); // 减少数量
129 surplusCount -= countToRemove;
130 excessValue += countToRemove * lowestArmor.Key.ItemValue;
131 Global.Debug($"Sold {countToRemove}x{lowestArmor.Key.ItemValue} from player's armory");
132 }
133
134 Global.Debug($"Sold {surplusCountCpy - surplusCount}x{equipmentAndThreshold.Key} items from player's armory");
135 }
136
137 Global.Debug($"Sold {excessValue} denars worth of equipment from player's armory");
138 return excessValue;
139 }

218 else { dict.Add(stringId, cnt); }
219 }
220
221 // 使用 Newtonsoft.Json 的 JsonSerializer 进行序列化
222 var json = JsonConvert.SerializeObject(dict, Formatting.Indented);
