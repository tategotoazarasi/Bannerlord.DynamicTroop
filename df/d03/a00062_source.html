38 public override void AfterStart() {
39 base.AfterStart();
40 Global.Log("AfterStart", Colors.Green, Level.Debug);
41 if (!Mission.DoesMissionRequireCivilianEquipment
42 && Mission.CombatType == Mission.MissionCombatType.Combat
43 && Campaign.Current.MainParty != null
44 && MapEvent.PlayerMapEvent != null) {
45 var tasks = new List<Task>();
46 Distributors.Add(Campaign.Current.MainParty.Id, new PartyEquipmentDistributor(Mission, Campaign.Current.MainParty, ArmyArmory.Armory));
47 foreach (var party in MapEvent.PlayerMapEvent.InvolvedParties) {
48 var mobileParty = party.MobileParty;
49 if (mobileParty != null && mobileParty != Campaign.Current.MainParty && mobileParty.IsValid()) {
50 var partyArmory = EveryoneCampaignBehavior.PartyArmories.TryGetValue(mobileParty.Id, out var armory) ? armory : new Dictionary<ItemObject, int>();
51 Distributors.Add(mobileParty.Id, new PartyEquipmentDistributor(Mission.Current, mobileParty, partyArmory));
52 }
53 }
54
55 foreach (var distributor in Distributors) { tasks.Add(Task.Run(() => { distributor.Value.RunAsync(); })); }
56
57 Task.WhenAll(tasks).GetAwaiter().GetResult();
58 Global.Debug("All distribution tasks finished");
59 }
60 }

69 public override void OnAgentRemoved(Agent affectedAgent, Agent affectorAgent, AgentState agentState, KillingBlow blow) {
70 if (Mission is { CombatType: Mission.MissionCombatType.Combat, PlayerTeam.IsValid: true } &&
71 agentState is AgentState.Killed or AgentState.Unconscious &&
72 affectedAgent.IsValid() &&
73 affectorAgent.IsValid() &&
74 !_processedAgents.Contains(affectedAgent) &&
75 affectedAgent.Character is { IsHero: false }) {
76 Global.Log($"agent {affectedAgent.Character.Name}#{affectedAgent.Index} removed", Colors.Green, Level.Debug);
77 _ = _processedAgents.Add(affectedAgent);
78 var affectedPartyId = Global.GetAgentParty(affectedAgent.Origin)?.Id;
79 var affectorPartyId = Global.GetAgentParty(affectorAgent.Origin)?.Id;
80 if (!affectedPartyId.HasValue || !affectorPartyId.HasValue) return;
81
82 if (!_partyBattleRecords.ContainsKey(affectedPartyId.Value))
83 _partyBattleRecords.TryAdd(affectedPartyId.Value, new PartyBattleRecord());
84
85 if (!_partyBattleRecords.ContainsKey(affectorPartyId.Value))
86 _partyBattleRecords.TryAdd(affectorPartyId.Value, new PartyBattleRecord());
87
88 var affectedBattleRecord = _partyBattleRecords[affectedPartyId.Value];
89 var affectorBattleRecord = _partyBattleRecords[affectorPartyId.Value];
90
91 // 获取受击部位
92 var hitBodyPart = blow.VictimBodyPart;
93
94 // 获取受击部位的护甲
95 var armors = affectedAgent.GetAgentArmors();
96 var hitArmor = ArmorSelector.GetRandomArmorByBodyPart(armors, hitBodyPart);
97
98 Global.ProcessAgentEquipment(affectedAgent,
99 item => {
100 if (hitArmor == null || item.StringId != hitArmor.StringId) {
101 affectedBattleRecord.AddItemToRecover(item);
102 var dropChance = _random.NextFloat(); // 生成一个0到1之间的随机数
103 if (dropChance > (SubModule.Settings?.DropRate ?? 1f)) return;
104
105 affectorBattleRecord.AddLootedItem(item);
106 Global.Log($"{item.StringId} looted", Colors.Green, Level.Debug);
107 }
108 else if (item.StringId == hitArmor.StringId) { Global.Log($"{item.StringId} damaged", Colors.Red, Level.Debug); }
109 });
110 }
111
112 base.OnAgentRemoved(affectedAgent, affectorAgent, agentState, blow);
113 }

151 private void OnMissionEnded(MissionResult? missionResult) {
152 Global.Log("OnMissionEnded() called", Colors.Green, Level.Debug);
153 if (_isMissionEnded) return;
154
155 _isMissionEnded = true;
156
157 var playerVictory = missionResult?.PlayerVictory ?? false;
158 var playerDefeat = missionResult?.PlayerDefeated ?? false;
159 var unresolvedBattle = missionResult == null || !missionResult.BattleResolved;
160
161 // 确定玩家方
162 var playerPartyId = Campaign.Current.MainParty.Id;
163
164 foreach (var kvPartyBattleSides in PartyBattleSides) {
165 var isPlayerParty = kvPartyBattleSides.Key == playerPartyId;
166 var isVictorious = !unresolvedBattle && (playerVictory && kvPartyBattleSides.Value == Mission.PlayerTeam.Side || playerDefeat && kvPartyBattleSides.Value != Mission.PlayerTeam.Side);
167 var isDefeated = !unresolvedBattle && (playerDefeat && kvPartyBattleSides.Value == Mission.PlayerTeam.Side || playerVictory && kvPartyBattleSides.Value != Mission.PlayerTeam.Side);
168
169 HandlePartyItems(kvPartyBattleSides.Key, isPlayerParty, isVictorious, isDefeated);
170 }
171
172 // 回收场上士兵的装备
173 foreach (var kvPartyBattleSides in PartyBattleSides) {
174 IEnumerable<Agent> partyAgents = Mission.Agents.WhereQ(agent => agent.IsValid() && agent.IsActive() && !agent.IsHero && Global.GetAgentParty(agent.Origin)?.Id == kvPartyBattleSides.Key);
175 ReturnEquipmentFromAgents(kvPartyBattleSides.Key, partyAgents);
176 }
177 }

186 private void HandlePartyItems(MBGUID partyId, bool isPlayerParty, bool isVictorious, bool isDefeated) {
187 // 检查是否存在对应的battleRecord
188 if (!_partyBattleRecords.TryGetValue(partyId, out var battleRecord)) {
189 Global.Warn($"No battle record found for party {partyId}");
190 return;
191 }
192
193 if (isVictorious) {
194 ReturnItemsToDestination(partyId, battleRecord.ItemsToRecover, isPlayerParty);
195 ReturnItemsToDestination(partyId, battleRecord.LootedItems, isPlayerParty);
196 if (isPlayerParty) {
197 MessageDisplayService.EnqueueMessage(new InformationMessage(LocalizedTexts.GetLootAddedMessage(battleRecord.LootedItemsCount), Colors.Green));
198 MessageDisplayService.EnqueueMessage(new InformationMessage(LocalizedTexts.GetItemsRecoveredFromFallenMessage(battleRecord.ItemsToRecoverCount), Colors.Green));
199 }
200 }
201 else if (!isDefeated) { ReturnItemsToDestination(partyId, battleRecord.ItemsToRecover, isPlayerParty); }

212 private void ReturnItemsToDestination(MBGUID partyId, ConcurrentDictionary<ItemObject, int> items, bool isPlayerParty) {
213 var totalItemCount = 0; // 用于累计物品总数
214
215 foreach (var item in items) {
216 if (item.Key == null || item.Value <= 0 || !ItemBlackList.Test(item.Key)) continue;
217
218 totalItemCount += item.Value;
219
220 if (isPlayerParty)
221 ArmyArmory.AddItemToArmory(item.Key, item.Value);
222 else if (Distributors.TryGetValue(partyId, out var distributor))
223 distributor.ReturnItem(item.Key, item.Value);
224 }
225
226 if (totalItemCount <= 0) return;
227
228 var partyType = isPlayerParty ? "Player party" : "Party";
229 Global.Log($"{partyType} {partyId} processed a total of {totalItemCount} items", Colors.Green, Level.Debug);
230 }

237 private void ReturnEquipmentFromAgents(MBGUID partyId, IEnumerable<Agent> agents) {
238 var isPlayerParty = partyId == Campaign.Current.MainParty.Id;
239 var totalItemCount = 0;
240
241 foreach (var agent in agents)
242 Global.ProcessAgentEquipment(agent,
243 item => {
244 if (item is null || !ItemBlackList.Test(item)) return;
245
246 totalItemCount++;
247 if (isPlayerParty)
248 ArmyArmory.AddItemToArmory(item);
249 else if (Distributors.TryGetValue(partyId, out var distributor))
250 distributor.ReturnItem(item, 1);
251 });
252
253 if (totalItemCount <= 0) return;
254
255 var partyType = isPlayerParty ? "Player party" : "Party";
256 Global.Log($"{partyType} {partyId} reclaimed {totalItemCount} items from agents", Colors.Green, Level.Debug);
257 }
