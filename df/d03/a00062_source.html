114 public int CompareTo(object? obj) {
115 if (obj == null) return 1;
116
117 if (obj is not Assignment other) throw new ArgumentException("Object is not an Assignment");
118
119 var tierComparison = Character.Tier.CompareTo(other.Character.Tier);
120 if (tierComparison != 0) return tierComparison;
121
122 // TroopType 比较，小的放在前面
123 var thisTroopType = Character.GetTroopType();
124 var otherTroopType = other.Character.GetTroopType();
125 var troopTypeComparison = thisTroopType.CompareTo(otherTroopType);
126 if (troopTypeComparison != 0) return troopTypeComparison;
127
128 // 按 IsRanged 比较，IsRanged 排在 !IsRanged 前面
129 if (!Character.IsRanged && other.Character.IsRanged) return 1;
130
131 if (Character.IsRanged && !other.Character.IsRanged) return -1;
132
133 // 按 IsMounted 比较，!IsMounted 排在 IsMounted 前面
134 if (Character.IsMounted && !other.Character.IsMounted) return 1;
135
136 if (!Character.IsMounted && other.Character.IsMounted) return -1;
137
138 var skillValueComparison = Character.SkillValue().CompareTo(other.Character.SkillValue());
139 if (skillValueComparison != 0) return skillValueComparison;
140
141 var equipmentValueComparison = Character.EquipmentValue().CompareTo(other.Character.EquipmentValue());
142 if (equipmentValueComparison != 0) return equipmentValueComparison;
143
144 var levelComparison = Character.Level.CompareTo(other.Character.Level);
145 if (levelComparison != 0) return levelComparison;
146
147 return 0; // 如果所有条件都相等，则认为两者相等
148 }

170 public void FillEmptySlots() {
171 foreach (var slot in Global.ArmourSlots) {
172 var referenceEquipment = ReferenceEquipment.GetEquipmentFromSlot(slot);
173 if (GetEquipmentFromSlot(slot) is not { IsEmpty: false, Item: not null }) {
174 var itemType = Helper.EquipmentIndexToItemEnumType(slot);
175 if (!itemType.HasValue) continue;

177 ItemObject? item;
178 if (referenceEquipment is { IsEmpty: false, Item: not null }) {

180 var itemsByCharacter = Cache.GetItemsByTypeTierAndCulture(itemType.Value, Character.Tier, Character.Culture);
181 if (itemsByCharacter != null) itemList.AddRange(itemsByCharacter);
182
183 if (referenceEquipment.Item.Culture is CultureObject cultureObject) {
184 var itemsByReference = Cache.GetItemsByTypeTierAndCulture(itemType.Value, (int)referenceEquipment.Item.Tier, cultureObject);
185 if (itemsByReference != null) itemList.AddRange(itemsByReference);
186 }
187
188 item = WeightedRandomSelector.SelectItem(itemList, referenceEquipment.Item.Effectiveness);
189 }
190 else {
191 item = Cache.GetItemsByTypeTierAndCulture(itemType.Value, Character.Tier, Character.Culture)?.GetRandomElement();
192 if (item == null) continue;
193 }
194
195 SetEquipment(slot, new EquipmentElement(item));
196 }
197 }
198
199 foreach (var slot in Global.EquipmentSlots)
200 if (GetEquipmentFromSlot(slot) is not { IsEmpty: false, Item: not null })
201 SetEquipment(slot, ReferenceEquipment.GetEquipmentFromSlot(slot));
