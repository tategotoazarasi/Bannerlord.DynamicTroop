57 private void OnDailyTick() {
58 var settings = ModSettings.Instance;
59 if (settings == null)
60 return;
61
62 // Only if Commander's Greed is OFF.
63 if (settings.CommandersGreed)
64 return;
65
66 var currentDayNumber = (int)CampaignTime.Now.ToDays;
67
68 // Run every 3 days.
69 if (_data.LastScrapDayNumber >= 0 && currentDayNumber - _data.LastScrapDayNumber < 3)
70 return;
71
72 _data.LastScrapDayNumber = currentDayNumber;
73
74 var cap = settings.ScrapCapPerCategory;
75 var targetCountPerCategory = cap - 1;
76
77 if (targetCountPerCategory <= 0)
78 return;
79
80 ScrapArmyArmoryByCategory(targetCountPerCategory);
81 }

84 private void Save() {
85 var i = ArmyArmory.Armory.GetEnumerator();
86 while (i.MoveNext())
87 if (i.Current is { IsEmpty: false, EquipmentElement: { IsEmpty: false, Item: not null }, Amount: > 0 }) {
88 if (!_data.Armory.ContainsKey(i.Current.EquipmentElement.Item.StringId))
89 _data.Armory.Add(i.Current.EquipmentElement.Item.StringId, i.Current.Amount);
90 else
91 _data.Armory[i.Current.EquipmentElement.Item.StringId] += i.Current.Amount;
92 }
93
94 i.Dispose();
95 }

97 private void Load(Data tempData) {
98 foreach (var item in tempData.Armory) {
99 var equipment = MBObjectManager.Instance.GetObject<ItemObject>(item.Key) ??
100 ItemObject.GetCraftedItemObjectFromHashedCode(item.Key);

102 _ = ArmyArmory.Armory.AddToCounts(equipment, item.Value);
103 else
104 Global.Warn($"cannot get object {item.Key}");
105 }

110 private static void ScrapArmyArmoryByCategory(int targetCountPerCategory) {
111 var itemsByType = new Dictionary<ItemObject.ItemTypeEnum, List<(EquipmentElement Equipment, int Amount)>>();
112
113 var enumerator = ArmyArmory.Armory.GetEnumerator();
114 while (enumerator.MoveNext()) {
115 var element = enumerator.Current;
116
117 if (element is not { IsEmpty: false, EquipmentElement: { IsEmpty: false, Item: not null }, Amount: > 0 })
118 continue;
119
120 var equipmentElement = element.EquipmentElement;
121 var itemType = equipmentElement.Item.ItemType;
122
123 if (!itemsByType.TryGetValue(itemType, out var list)) {
124 list = new List<(EquipmentElement Equipment, int Amount)>();
125 itemsByType.Add(itemType, list);
126 }
127
128
129 list.Add((equipmentElement, element.Amount));
130 }
131
132 enumerator.Dispose();
133
134 foreach (var kvp in itemsByType) {
135 var entries = kvp.Value;
136
137 var totalCount = 0;
138 for (var i = 0; i < entries.Count; i++)
139 totalCount += entries[i].Amount;
140
141 if (totalCount <= targetCountPerCategory)
142 continue;
143
144 var removeNeeded = totalCount - targetCountPerCategory;
145
146 // lowest value first
147 entries.Sort((a, b) => {
148 var valueCompare = a.Equipment.ItemValue.CompareTo(b.Equipment.ItemValue);
149 if (valueCompare != 0)
150 return valueCompare;
151
152 var itemCompare = string.CompareOrdinal(a.Equipment.Item.StringId, b.Equipment.Item.StringId);
153 return itemCompare != 0
154 ? itemCompare
155 : string.CompareOrdinal(a.Equipment.ItemModifier?.StringId, b.Equipment.ItemModifier?.StringId);
156 });
157
158 for (var i = 0; i < entries.Count && removeNeeded > 0; i++) {
159 (var equipment, var amount) = entries[i];
160 var removeCount = Math.Min(amount, removeNeeded);
161
162 ArmyArmory.Armory.AddToCounts(equipment, -removeCount);
163 removeNeeded -= removeCount;
164 }
165 }
166 }
