20 public override void SyncData(IDataStore dataStore) {
21 //InformationManager.DisplayMessage(new InformationMessage("Sync Data called", Colors.Green));
22 if (dataStore.IsSaving) {
23 _data.Armory.Clear();
24 Save();
25 var tempData = _data;
26 if (dataStore.SyncDataAsJson("DynamicTroopArmyArmory", ref tempData) && tempData != null) {
27 _data = tempData;
28 _data.Armory.Clear();
29 }
30 else { Global.Error("null data on save"); }
31 }
32 else if (dataStore.IsLoading) {
33 _data.Armory.Clear();
34 ArmyArmory.Armory.Clear();
35 var tempData = _data;
36 if (dataStore.SyncDataAsJson("DynamicTroopArmyArmory", ref tempData) && tempData != null) {
37 Load(tempData);
38 _data = tempData;
39 _data.Armory.Clear();
40 }
41 else { Global.Error("null data on load"); }
42 }
43 }

55 private void Save() {
56 var i = ArmyArmory.Armory.GetEnumerator();
57 while (i.MoveNext())
58 if (i.Current is { IsEmpty: false, EquipmentElement: { IsEmpty: false, Item: not null }, Amount: > 0 }) {
59 if (!_data.Armory.ContainsKey(i.Current.EquipmentElement.Item.StringId))
60 _data.Armory.Add(i.Current.EquipmentElement.Item.StringId, i.Current.Amount);
61 else
62 _data.Armory[i.Current.EquipmentElement.Item.StringId] += i.Current.Amount;
63 }
64
65 i.Dispose();
66 }
67
68 private void Load(Data tempData) {
69 foreach (var item in tempData.Armory) {
70 var equipment = MBObjectManager.Instance.GetObject<ItemObject>(item.Key) ??
71 ItemObject.GetCraftedItemObjectFromHashedCode(item.Key);
72 if (equipment != null && item.Value > 0)
73 _ = ArmyArmory.Armory.AddToCounts(equipment, item.Value);
74 else
75 Global.Warn($"cannot get object {item.Key}");
76 }
77
78 Global.Debug($"loaded {tempData.Armory.Count} entries for player");
79 }
