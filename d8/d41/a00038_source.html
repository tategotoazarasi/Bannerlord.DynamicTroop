18 private static void Prefix(PartyBase winnerParty, ItemRoster gainedLoots) {

20 return;

24 if (mapEvent == null ||

33 if (float.IsNaN(playerContribution) || float.IsInfinity(playerContribution))
34 playerContribution = 0f;
35
36 playerContribution = MBMath.ClampFloat(
37 playerContribution * (ModSettings.Instance?.DropRate ?? 1f),
38 0f,
39 1f);
40
41 foreach (var defeatedParty in mapEvent.PartiesOnSide(mapEvent.DefeatedSide)) {
42 var defeatedMobileParty = defeatedParty.Party.MobileParty;

44 continue;

46 var defeatedArmory = EveryoneCampaignBehavior.SanitizePartyArmory(defeatedMobileParty.Id);
47 if (defeatedArmory == null)
48 continue;
49
50 foreach (var entry in defeatedArmory) {
51 if (entry.Value <= 0 ||
52 !ArmyArmory.TryResolveArmoryItem(entry.Key, out var item) ||
53 !ItemBlackList.Test(item))
54 continue;
55
56 var expectedCount = entry.Value * playerContribution;
57 var lootCount = (int)expectedCount;
58 if (_random.NextDouble() < expectedCount - lootCount)
59 lootCount++;

61 if (lootCount > 0)

63 }
64 }
65

69 }

72 var item = rosterElement.EquipmentElement.Item;

74 dynamicTroopLoot.AddToCounts(rosterElement.EquipmentElement, rosterElement.Amount);
75 }
76
77 ReplaceRosterContents(gainedLoots, dynamicTroopLoot);
78 }

80 private static ItemRoster CreateSanitizedRoster(ItemRoster sourceRoster) {
