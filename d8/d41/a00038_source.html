151 public static int CalculateWeaponTierBonus(this ItemObject? weapon, bool mounted) {
152 if (weapon == null || mounted) return 0; // 如果骑马，则不应用任何加成
153
154 var bonus = 0;
155 MBReadOnlyList<WeaponComponentData> weaponFlags = weapon.WeaponComponent.Weapons;
156 var weaponFlag = (WeaponFlags)weaponFlags.Aggregate(0u, (current, flag) => current | (uint)flag.WeaponFlags);
157
158 // 为每个匹配的WeaponFlag增加加成
159 if (weaponFlag.HasFlag(WeaponFlags.BonusAgainstShield)) bonus++;
160
161 if (weaponFlag.HasFlag(WeaponFlags.CanKnockDown)) bonus++;
162
163 if (weaponFlag.HasFlag(WeaponFlags.CanDismount)) bonus++;
164
165 if (weaponFlag.HasFlag(WeaponFlags.MultiplePenetration)) bonus++;
166
167 if (weapon.IsBracable()) bonus++;
168
169 return bonus;
170 }

184 public static int CompareArmor(this ItemObject? item, ItemObject? other) {
185 // null 永远排在非 null 前面
186 if (item == null && other != null) return -1;
187
188 if (item != null && other == null) return 1;
189
190 if (item == null && other == null) return 0;
191
192 if (item != null && other != null) {
193 // Tier 低的排在 Tier 高的前面
194 var tierComparison = item.Tier.CompareTo(other.Tier);
195 if (tierComparison != 0) return tierComparison;
196
197 // HasArmorComponent 为 false 的排在 HasArmorComponent 为 true 的前面
198 if (!item.HasArmorComponent && other.HasArmorComponent) return -1;
199
200 if (item.HasArmorComponent && !other.HasArmorComponent) return 1;
201
202 // 如果两者都 HasArmorComponent，则调用 ArmorComponent.MaterialType 的 CompareTo 方法
203 if (item.HasArmorComponent && other.HasArmorComponent) {
204 var materialComparison = item.ArmorComponent.MaterialType.CompareTo(other.ArmorComponent.MaterialType);
205 if (materialComparison != 0) return materialComparison;
206 }
207
208 // Value 低的排在 Value 高的前面
209 var valueComparison = item.Value.CompareTo(other.Value);
210 if (valueComparison != 0) return valueComparison;
211
212 // Culture 为 null 的排在 Culture 不为 null 的前面
213 if (item.Culture == null && other.Culture != null) return -1;
214
215 if (item.Culture != null && other.Culture == null) return 1;
216
217 return 0; // 如果所有条件都相等，则认为两者相等
218 }
219
220 return 0;
221 }
