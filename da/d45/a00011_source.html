156 private void OnDailyTickTown(Town town) {
157 var ownerClan = town.OwnerClan;
158
159 var sourceSettlement = town.Settlement;
160 if (sourceSettlement == null)
161 return;
162
163 if (_activeCaravanBySourceSettlementId.TryGetValue(sourceSettlement.Id, out var existingCaravanId)) {
164 var existingCaravan = FindMobilePartyById(existingCaravanId);
165 if (existingCaravan != null && existingCaravan.IsActive)
166 return;
167
168 // Keep it clean
169 _activeCaravanBySourceSettlementId.Remove(sourceSettlement.Id);
170 }
171
172
173 if (ownerClan == null || ownerClan == Clan.PlayerClan)
174 return;
175
176 var prosperity = town.Prosperity;
177
178 if (prosperity < MIN_PROSPERITY)
179 return;
180
181 var chance = GetDailySpawnChance(prosperity);
182
183 if (MBRandom.RandomFloat > chance)
184 return;
185
186 var targetParty = FindTargetLordParty(ownerClan);
187
188 if (targetParty == null)
189 return;
190
191 SpawnReinforcementCaravan(town, ownerClan, targetParty);
192 }

195 private void OnTick(float dt) {
196 _escortRefreshTimer += dt;
197 if (_escortRefreshTimer < ESCORT_REFRESH_SECONDS) { return; }
198
199 _escortRefreshTimer = 0f;
200
201 // take a snapshot to avoid modifying the dictionary during iteration
202 foreach (var pair in _targetByCaravanId.ToList()) {
203 var caravanParty = FindMobilePartyById(pair.Key);
204 if (caravanParty == null || !caravanParty.IsActive) {
205 _targetByCaravanId.Remove(pair.Key);
206 ReinforcementCaravanIds.Remove(pair.Key);
207 continue;
208 }
209
210 if (TryRunAwayFromNearbyThreat(caravanParty)) { continue; }
211
212 var targetParty = FindMobilePartyById(pair.Value);
213
214
215 // guarantee target
216 if (targetParty == null || !targetParty.IsActive) {
217 if (_ownerClanByCaravanId.TryGetValue(caravanParty.Id, out var ownerClanId)) {
218 var ownerClan = FindClanById(ownerClanId);
219 if (ownerClan != null) {
220 var newTarget = FindTargetLordParty(ownerClan, pair.Value);
221 if (newTarget != null) {
222 _targetByCaravanId[caravanParty.Id] = newTarget.Id;
223 targetParty = newTarget;
224 }
225 else { _returningCaravanIds.Add(caravanParty.Id); }
226 }
227 else { _returningCaravanIds.Add(caravanParty.Id); }
228 }
229 else { _returningCaravanIds.Add(caravanParty.Id); }
230 }
231
232 // return and despawn
233 if (_returningCaravanIds.Contains(caravanParty.Id)) {
234 if (_sourceSettlementByCaravanId.TryGetValue(caravanParty.Id, out var sourceSettlementId)) {
235 var sourceSettlement = FindSettlementById(sourceSettlementId);
236 if (sourceSettlement != null) {
237 // check twice
238 if (caravanParty.CurrentSettlement != null && caravanParty.CurrentSettlement != sourceSettlement) { LeaveSettlementAction.ApplyForParty(caravanParty); }
239
240 caravanParty.Ai.SetDoNotMakeNewDecisions(true);
241
242 caravanParty.SetMoveGoToSettlement(sourceSettlement, MobileParty.NavigationType.All, false);
243 caravanParty.RecalculateShortTermBehavior();
244
245 if (caravanParty.CurrentSettlement == sourceSettlement) { CleanupCaravan(caravanParty); }
246 }
247 else { CleanupCaravan(caravanParty); }
248 }
249 else { CleanupCaravan(caravanParty); }
250
251 continue;
252 }
253
254 // when targetparty is ensured
255 if (targetParty == null) {
256 _returningCaravanIds.Add(caravanParty.Id);
257 continue;
258 }
259
260 var stashReceiverParty = targetParty;
261
262 var movementTargetParty = targetParty;
263 if (movementTargetParty.Army != null && movementTargetParty.Army.LeaderParty != null) { movementTargetParty = movementTargetParty.Army.LeaderParty; }
264
265 // if they're inside a settlement
266 var targetSettlement = movementTargetParty.CurrentSettlement;
267 if (targetSettlement != null) {
268 if (caravanParty.CurrentSettlement != null && caravanParty.CurrentSettlement != targetSettlement) { LeaveSettlementAction.ApplyForParty(caravanParty); }
269
270 caravanParty.Ai.SetDoNotMakeNewDecisions(true);
271
272
273 caravanParty.SetMoveGoToSettlement(targetSettlement, MobileParty.NavigationType.All, false);
274 caravanParty.RecalculateShortTermBehavior();
275
276 if (caravanParty.CurrentSettlement == targetSettlement) {
277 TransferCargoToTargetStash(caravanParty, stashReceiverParty);
278 CleanupCaravan(caravanParty);
279 }
280
281 continue;
282 }
283
284 // for sieges/battles, keep the distance
285 if (movementTargetParty.MapEvent != null || movementTargetParty.SiegeEvent != null) {
286 if (caravanParty.CurrentSettlement != null) { LeaveSettlementAction.ApplyForParty(caravanParty); }
287
288 caravanParty.Ai.SetDoNotMakeNewDecisions(true);
289
290 var delta = caravanParty.GetPosition2D - movementTargetParty.GetPosition2D;
291 if (delta.LengthSquared <= WAIT_DISTANCE_SQUARED) { caravanParty.SetMoveModeHold(); }
292 else {
293 var waitPoint = GetWaitPointNear(caravanParty, movementTargetParty);
294 caravanParty.SetMoveGoToPoint(waitPoint, MobileParty.NavigationType.Default);
295 }
296
297 caravanParty.RecalculateShortTermBehavior();
298 continue;
299 }
300
301
302 // regularly
303 if (caravanParty.CurrentSettlement != null) { LeaveSettlementAction.ApplyForParty(caravanParty); }
304
305 caravanParty.Ai.SetDoNotMakeNewDecisions(true);
306
307 var destination = new CampaignVec2(movementTargetParty.Position.ToVec2(), movementTargetParty.Position.IsOnLand);
308 caravanParty.SetMoveGoToPoint(destination, MobileParty.NavigationType.Default);
309 caravanParty.RecalculateShortTermBehavior();
310
311 if (HasArrived(caravanParty, movementTargetParty)) {
312 TransferCargoToTargetStash(caravanParty, stashReceiverParty);
313 CleanupCaravan(caravanParty);
314 }
315 }

335 var chance = BASE_CHANCE_AT_MIN_PROSPERITY + extraProsperity * CHANCE_PER_PROSPERITY_POINT;

340 MobileParty? bestParty = null;
341 var bestDistance = int.MaxValue;
342
343 foreach (var party in Campaign.Current.MobileParties) {
344 if (party == null || !party.IsActive)
345 continue;
346
347 var leaderHero = party.LeaderHero;

349 continue;
350
351 // lords
352 if (party.IsCaravan || party.IsVillager || party.IsMilitia || party.IsBandit)
353 continue;
354
355 var partySize = party.MemberRoster.TotalManCount;
356 var distance = Math.Abs(partySize - TARGET_PARTY_SIZE);
357
358 if (distance < bestDistance) {
359 bestDistance = distance;
360 bestParty = party;
361 }
362 }
363
364 return bestParty;

436 private static CampaignVec2 GetWaitPointNear(MobileParty caravanParty, MobileParty movementTargetParty) {
437 var targetPos = movementTargetParty.Position.ToVec2();
438 var caravanPos = caravanParty.Position.ToVec2();
439
440 var away = caravanPos - targetPos;
441
442 if (away.LengthSquared < 0.0001f) { away = new Vec2(1f, 0f); }
443 else { away = away.Normalized(); }
444
445 var waitPos = targetPos + away * WAIT_DISTANCE;
446 return new CampaignVec2(waitPos, movementTargetParty.Position.IsOnLand);
447 }

450 private static bool TryRunAwayFromNearbyThreat(MobileParty caravanParty) {
451 MobileParty? nearestThreat = null;
452 var nearestThreatDistanceSquared = float.MaxValue;
453
454 foreach (var otherParty in Campaign.Current.MobileParties) {
455 if (otherParty == null || !otherParty.IsActive)
456 continue;
457
458 if (otherParty == caravanParty)
459 continue;
460
461 if (!caravanParty.MapFaction.IsAtWarWith(otherParty.MapFaction))
462 continue;
463
464 var delta = caravanParty.GetPosition2D - otherParty.GetPosition2D;
465 var distanceSquared = delta.LengthSquared;
466
467 if (distanceSquared > THREAT_SCAN_RADIUS_SQUARED)
468 continue;
469
470 if (distanceSquared < nearestThreatDistanceSquared) {
471 nearestThreatDistanceSquared = distanceSquared;
472 nearestThreat = otherParty;
473 }
474 }
475
476 if (nearestThreat == null)
477 return false;
478
479 var caravanManCount = caravanParty.MemberRoster.TotalManCount;
480 var enemyManCount = nearestThreat.MemberRoster.TotalManCount;
481
482 if (enemyManCount <= caravanManCount * FLEE_IF_ENEMY_STRENGTH_MULTIPLIER)
483 return false;
484
485
486 var away = caravanParty.GetPosition2D - nearestThreat.GetPosition2D;
487 if (away.LengthSquared < 0.0001f)
488 away = new Vec2(1f, 0f);
489 else
490 away = away.Normalized();
491
492 var fleePos = caravanParty.GetPosition2D + away * FLEE_POINT_DISTANCE;
493 caravanParty.SetMoveGoToPoint(new CampaignVec2(fleePos, caravanParty.Position.IsOnLand), MobileParty.NavigationType.Default);
494 caravanParty.RecalculateShortTermBehavior();
495
496 return true;
497 }

630 private static void TransferCargoToTargetStash(MobileParty caravanParty, MobileParty targetParty) {
631 if (!EveryoneCampaignBehavior.PartyArmories.TryGetValue(targetParty.Id, out var targetArmory)) {
632 targetArmory = new Dictionary<ItemObject, int>();
633 EveryoneCampaignBehavior.PartyArmories.Add(targetParty.Id, targetArmory);
634 }
635
636 var roster = caravanParty.ItemRoster;
637 var enumerator = roster.GetEnumerator();
638
639 while (enumerator.MoveNext()) {
640 var element = enumerator.Current;
641
642 if (element is not { IsEmpty: false, EquipmentElement: { IsEmpty: false, Item: not null }, Amount: > 0 })
643 continue;
644
645 var item = element.EquipmentElement.Item;
646
647 if (!targetArmory.TryGetValue(item, out var currentAmount))
648 targetArmory.Add(item, element.Amount);
649 else
650 targetArmory[item] = currentAmount + element.Amount;
651 }
652
653 enumerator.Dispose();
654 }
