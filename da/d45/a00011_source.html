153 private void GarbageCollectEquipments(MobileParty mobileParty) {
154 if (!mobileParty.IsValid() || mobileParty.MemberRoster?.GetTroopRoster() == null) return;
155
156 var memberCnt = mobileParty.MemberRoster.GetTroopRoster()
157 .WhereQ(element => element.Character is { IsHero: false })
158 .SumQ(element => element.Number);
159 foreach (var equipmentAndThreshold in EquipmentAndThresholds) {
160 var partyArmory = PartyArmories.GetValueSafe(mobileParty.Id);
161 if (partyArmory == null || partyArmory.IsEmpty()) return;
162
163 var armorTotalCount = partyArmory.WhereQ(kv => kv.Key.ItemType == equipmentAndThreshold.Key)
164 .SumQ(kv => kv.Value);
165 var surplusCount = armorTotalCount - memberCnt;
166 if (surplusCount <= 0) continue;
167
168 var surplusCountCpy = surplusCount;
169
170 // 创建优先级队列
171 ItemPriorityQueue armorQueue = new(new EquipmentEffectivenessComparer());
172 foreach (var kv in partyArmory.WhereQ(kv => kv.Key.ItemType == equipmentAndThreshold.Key))
173 armorQueue.Enqueue(kv.Key, ((int)kv.Key.Tier, kv.Key.Value));
174
175 // 移除多余的装备
176 while (surplusCount > 0 && armorQueue.Count > 0) {
177 var lowestArmor = armorQueue.Dequeue();
178 var countToRemove = Math.Min(partyArmory[lowestArmor.Key], surplusCount);
179 partyArmory[lowestArmor.Key] -= countToRemove;
180 surplusCount -= countToRemove;
181 if (partyArmory[lowestArmor.Key] == 0) _ = partyArmory.Remove(lowestArmor.Key);
182 }
183
184 Global.Debug($"Garbage collected {surplusCountCpy - surplusCount}x{equipmentAndThreshold.Key} armors from party {mobileParty.Name}");
185 }
186 }

372 continue;
373
374 itemsWithCount[item.Key] = itemsWithCount.TryGetValue(item.Key, out var currentCount)
375 ? currentCount + item.Value

439 public static Dictionary<ItemObject, int>? SanitizePartyArmory(MBGUID partyId) {
440 if (!PartyArmories.TryGetValue(partyId, out var armory))
441 return null;
442
443 var sanitizedItems = new Dictionary<ItemObject, int>();
444 foreach (var entry in armory) {
445 if (entry.Value <= 0 || !ArmyArmory.TryResolveArmoryItem(entry.Key, out var item))
446 continue;
447
448 sanitizedItems[item] = sanitizedItems.TryGetValue(item, out var currentCount)
449 ? currentCount + entry.Value
450 : entry.Value;
451 }
452
453 armory.Clear();
454 foreach (var entry in sanitizedItems)
455 armory[entry.Key] = entry.Value;
456
457 return armory;
458 }

492 continue;
493

495 ? existingItems

498 foreach (var itemEntry in partyEntry.Value) {

502 serializedItems[item.StringId] = serializedItems.TryGetValue(item.StringId, out var currentCount)
503 ? currentCount + itemEntry.Value

505 }
506
507 if (serializedItems.Count > 0)
508 serializedArmories[partyEntry.Key.InternalValue] = serializedItems;

514 private void RestoreReadyPartyArmories() {
515 foreach (var pendingParty in new Dictionary<uint, Dictionary<string, int>>(_unresolvedPartyArmories)) {
516 var partyId = new MBGUID(pendingParty.Key);
517 if (FindActiveParty(partyId) == null)
518 continue;
519

521 armory = new Dictionary<ItemObject, int>();
522 PartyArmories[partyId] = armory;
523 }
524
525 foreach (var pendingItem in new Dictionary<string, int>(pendingParty.Value)) {
526 var item = ArmyArmory.ResolveArmoryItem(pendingItem.Key);
527 if (item == null)
528 continue;
529
530 armory[item] = armory.TryGetValue(item, out var currentCount)
531 ? currentCount + pendingItem.Value
532 : pendingItem.Value;
533 pendingParty.Value.Remove(pendingItem.Key);
534 }
535
536 if (pendingParty.Value.Count == 0)
537 _unresolvedPartyArmories.Remove(pendingParty.Key);
538 else
539 _unresolvedPartyArmories[pendingParty.Key] = pendingParty.Value;
