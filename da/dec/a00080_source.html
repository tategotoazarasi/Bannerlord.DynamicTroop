57 private static bool IsCombatMissionWithValidData(Mission mission, AgentBuildData agentBuildData) {

102 private static void EnsureDistributorExists(DynamicTroopMissionLogic missionLogic, MobileParty party) {
103 if (missionLogic.Distributors.ContainsKey(party.Id)) return;

105 var partyArmory = EveryoneCampaignBehavior.PartyArmories.TryGetValue(party.Id, out var armory) ? armory : new Dictionary<ItemObject, int>();
