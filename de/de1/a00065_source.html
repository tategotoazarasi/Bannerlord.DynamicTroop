96 private static int SellExcessEquipment() {
97 RebuildArmory();
98 var excessValue = 0;
99 var playerParty = MobileParty.MainParty;
100 if (playerParty?.MemberRoster?.GetTroopRoster() == null) return 0;
101
102 var memberCnt = playerParty.MemberRoster.GetTroopRoster()
103 .WhereQ(element => element.Character is { IsHero: false })
104 .SumQ(element => element.Number);
105
106 foreach (var equipmentAndThreshold in EveryoneCampaignBehavior.EquipmentAndThresholds) {
107 var armorTotalCount = Armory.WhereQ(kv => kv.EquipmentElement.Item?.ItemType == equipmentAndThreshold.Key)
108 .SumQ(kv => kv.Amount);
109 var surplusCount = armorTotalCount - equipmentAndThreshold.Value(memberCnt);
110 if (surplusCount <= 0) continue;
111
112 var surplusCountCpy = surplusCount;
113
114 // 创建优先级队列
115 ItemPriorityQueue armorQueue = new(new ArmorElementComparer());
116 foreach (var kv in Armory.WhereQ(kv => kv.EquipmentElement.Item?.ItemType == equipmentAndThreshold.Key))
117 armorQueue.Enqueue(kv.EquipmentElement, kv.EquipmentElement.ItemValue);
118
119 // 移除多余的装备
120 while (surplusCount > 0 && !armorQueue.IsEmpty) {
121 var lowestArmor = armorQueue.Dequeue();
122 var countToRemove = Math.Min(Armory.GetElementNumber(Armory.FindIndexOfElement(lowestArmor.Key)),
123 surplusCount);
124
125 //Global.Debug($"countToRemove={countToRemove}, lowestArmorNumber={Armory.GetItemNumber(lowestArmor.Key)}");
126 _ = Armory.AddToCounts(lowestArmor.Key, -countToRemove); // 减少数量
127 surplusCount -= countToRemove;
128 excessValue += countToRemove * lowestArmor.Key.ItemValue;
129 Global.Debug($"Sold {countToRemove}x{lowestArmor.Key.ItemValue} from player's armory");
130 }
131
132 Global.Debug($"Sold {surplusCountCpy - surplusCount}x{equipmentAndThreshold.Key} items from player's armory");
133 }
134
135 Global.Debug($"Sold {excessValue} denars worth of equipment from player's armory");
136 return excessValue;
137 }
