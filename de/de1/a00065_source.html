115 public static bool Test(ItemObject item) {
116 try {
117 if (Cache.TryGetValue(item, out var result)) { return result; }
118
119 if (item is not { StringId: not null, Name: not null } || item.StringId.IsEmpty() || item.Name.ToString().IsEmpty()) return true;
120 var stringId = item.StringId;
121 var name = item.Name.ToString();
122
123 result = !StringIds.Contains(stringId) && !Names.Contains(name) && !StringIdPatterns.Any(pattern => Regex.IsMatch(stringId, pattern)) && !NamePatterns.Any(pattern => Regex.IsMatch(name, pattern));
124 Cache.Add(item, result);
125 return result;
126 }
127 catch (Exception e) {
128 Global.Error(e.Message);
129 return false;
130 }
