120 public static bool Test(ItemObject item) {
121 try {
122 if (Cache.TryGetValue(item, out var result)) { return result; }
123
124 if (item is not { StringId: not null, Name: not null } || item.StringId.IsEmpty() || item.Name.ToString().IsEmpty()) return true;

127
128 result = !StringIds.Contains(stringId) && !Names.Contains(name) && !StringIdPatterns.Any(pattern => pattern.IsMatch(stringId)) && !NamePatterns.Any(pattern => pattern.IsMatch(name));
