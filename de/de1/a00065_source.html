121 public static bool Test(ItemObject item) {
122 try {
123 if (Cache.TryGetValue(item, out var result)) { return result; }
124
125 if (item is not { StringId: not null, Name: not null } || item.StringId.IsEmpty() || item.Name.ToString().IsEmpty()) return true;
126 var stringId = item.StringId;
127 var name = item.Name.ToString();
128
129 // 并行执行所有匹配检查
130 var isBlacklisted = new[] {
131 // StringIds 和 Names 检查可以直接并行
132 Task.Run(() => !StringIds.Contains(stringId)),
133 Task.Run(() => !Names.Contains(name)),
134 // 正则表达式检查需要更细粒度的并行操作
135 Task.Run(() => !StringIdPatterns.AsParallel().Any(pattern => pattern.IsMatch(stringId))),
136 Task.Run(() => !NamePatterns.AsParallel().Any(pattern => pattern.IsMatch(name)))
137 };
138
139 // 等待所有任务完成，并检查所有条件是否满足
140 result = Task.WhenAll(isBlacklisted).Result.All(matchResult => matchResult);
141
142 Cache[item] = result;
143 return result;
144 }
145 catch (Exception e) {
146 Global.Error(e.Message);
147 return false;
148 }
149 }
